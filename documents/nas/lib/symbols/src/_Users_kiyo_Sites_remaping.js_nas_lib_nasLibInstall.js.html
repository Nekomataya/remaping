 91 /**
 92  * @param col
 93  * @param line
 94  * @param width
 95  * @param height
 96  * @returns {*[]}
 97  */
 98 function nasGrid(col, line, width, height) {
 99     left = (col * colUnit) + leftMargin + leftPadding;
100     top = (line * lineUnit) + topMargin + topPadding;
101     right = left + width - rightPadding;
102     bottom = (height <= lineUnit) ? top + height - bottomPadding - quartsOffset : top + height - bottomPadding;
103     return [left, top, right, bottom];
104 }

218 nas_Installer.pathReplace = function (myString) {
219     myString = myString.replace(/\%STARTUP\%/g, this.replacePath.startup);
220     myString = myString.replace(/\%NAS\%/g, this.replacePath.nas);
221     myString = myString.replace(/\%INSTALL\%/g, this.replacePath.install);
222     myString = myString.replace(/\%SOURCE\%/g, this.replacePath.source);
223     myString = myString.replace(/\%USER\%/g, this.replacePath.user);
224     return myString;

269     for (var idx = 0; idx < this.myInstall.length; idx++) {
270 
271         var myAction = "NOP";
272         if (this.myInstall[idx][0] instanceof Function) {
273             myAction = "doFunction"
274         } else {
275             switch (this.myInstall[idx][0]) {
276                 case    "mkdir":
277                     myAction = "makeFolder";
278                     break;
279                 case    "rmdir":
280                     myAction = "removeFolder";
281                     break;
282                 case    "cd":
283                     myAction = "changeDir";
284                     break;
285                 case    "confirm":
286                     myAction = "confirm";
287                     break;
288                 case    "shutdown":
289                     myAction = "shutdown";
290                     break;
291                 case    "cp":
292                     myAction = "copyFile";
293                     break;
294                 case    "mv":
295                     myAction = "renameFile";
296                     break;
297                 case    "rm":
298                     myAction = "deleteFile";
299                     break;
300                 case    "clearAll":
301                     myAction = "clearSelfFiles";
302                     break;
303                 default    :
304                     if (this.myInstall[idx].length = 2) {
305                         myAction = "copyFile";
306                     }
307             }
308         }
309         switch (myAction) {
310             case    "confirm":
311                 if (!confirm(this.pathReplace(this.myInstall[idx][1]))) {
312                     return false;
313                 }
314                 break;
315             case    "shutdown":
316                 nas_Installer.isShutdown = true;
317                 return false;
318                 break;
319             case    "makeFolder":
320                 var targetDir = new Folder(this.pathReplace(this.myInstall[idx][1]));
321                 if (!targetDir.exists) {
322                     if (this.debug) {
323                         alert(targetDir.fsName + " を作った気分");
324                     } else {
325                         try {
326                             targetDir.create();
327                         } catch (err) {
328                             this.abortProcess(myAction + ":\n" + err);
329                             return false;
330                         }
331                     }
332                     this.pushLog(myAction + " : " + targetDir.fsName);
333                 } else {
334                     this.showMsg("フォルダ " + targetDir.fsName + " がすでにあるっぽいのでパス");
335                     this.pushLog(myAction + " : " + targetDir.fsName + "allready exists");
336                 }
337                 break;
338             case    "removeFolder":
339                 var targetDir = new Folder(this.pathReplace(this.myInstall[idx][1]));
340                 if (targetDir.exists) {
341                     if (this.debug) {
342                         if (targetDir.getFiles().length != 0) {
343                             this.showMsg("フォルダが空でないけど " + targetDir.fsName + " を削除した気分");
344                         } else {
345                             this.showMsg(targetDir.fsName + " を削除した気分");
346                         }
347                     } else {
348                         /**
349                          * @desc フォルダコンテンツ確認
350                          */
351                         if (targetDir.getFiles().length != 0) {
352                             this.abortProcess(myAction + ":\nフォルダが空でない様");
353                             return false;
354                         } else {
355                             try {
356                                 targetDir.remove();
357                             } catch (err) {
358                                 this.abortProcess(myAction + ":\n" + err);
359                                 return false;
360                             }
361                         }
362                     }
363                     this.pushLog(myAction + " : " + targetDir.fsName);
364                 } else {
365                     this.showMsg("フォルダ " + targetDir.fsName + " は無い様");
366                     this.pushLog(myAction + " : " + targetDir.fsName + "cannot remove");
367                 }
368                 break;
369             case    "changeDir":
370                 if (!this.myInstall[idx][1]) {
371                     Folder.current = new Folder(this.replacePath.source);//ソースフォルダに戻る
372                 } else {
373                     Folder.current = new Folder(this.myInstall[idx][1]);
374                 }
375                 this.pushLog(myAction + " : " + Folder.current.fsName);
376                 break;
377             case    "copyFile":
378                 if (this.myInstall[idx][0] == "cp") {
379                     var destFile = new File(this.pathReplace(this.myInstall[idx][1]));
380                     var targetFile = new File(this.pathReplace(this.myInstall[idx][2]));
381                 } else {
382                     var destFile = new File(this.pathReplace(this.myInstall[idx][0]));
383                     var targetFile = new File(this.pathReplace(this.myInstall[idx][1]));
384                 }
385                 if ((destFile.exists) && (!targetFile.exists)) {
386                     if (destFile.name.match(/\.jsx?$/)) {
387                         this.pushLog(myAction);
388                         this.pushLog("file copy with WordReplace :");
389 
390                         //include 置換コピー
391                         if (this.debug) {
392                             alert(destFile.fsName + " を " + targetFile.fsName + " に置換コピーした気分");
393                         } else {
394                             try {
395                                 if (!this.copyScriptWithReplace(destFile, targetFile)) {
396                                     this.abortProcess(myAction + ":\n" + err);
397                                     return false;
398                                 }
399 
400                             } catch (err) {
401                                 this.abortProcess(myAction + ":\n" + err);
402                                 return false;
403                             }
404                         }
405                     } else {
406 
407                         //単純ファイルコピー
408                         if (this.debug) {
409                             alert(destFile.fsName + " を " + targetFile.fsName + " にコピーした気分");
410                         } else {
411                             try {
412                                 destFile.copy(targetFile)
413                             } catch (err) {
414                                 this.abortProcess(myAction + ":\n" + err);
415                                 return false;
416                             }
417                         }
418                         this.pushLog("file copy :" + destFile.fsName + " > " + targetFile.fsName);
419                     }
420                 } else {
421                     this.showMsg("ファイル\n" + destFile.fsName + " : " + targetFile.fsName + "\nを確認しましょう。");
422                 }
423                 break;
424             case    "renameFile":
425                 var destFile = new File(this.pathReplace(this.myInstall[idx][1]));
426                 var targetFile = new File(this.pathReplace(this.myInstall[idx][2]));
427 
428                 if ((destFile.exists) && (!targetFile.exists)) {
429 
430                     //ファイルをコピーして元ファイルを消去
431                     if (this.debug) {
432                         alert(destFile.fsName + " を " + targetFile.fsName + " に移動した気分");
433                     } else {
434                         //Adobeスクリプトにmoveメソッドは無いので、copy + remove にする。
435                         try {
436                             if (destFile.copy(targetFile)) {
437                                 destFile.remove();
438                             }
439                         } catch (err) {
440                             this.abortProcess(myAction + ":\n" + err);
441                             return false;
442                         }
443                     }
444                     this.pushLog(myAction + " :" + destFile.fsName + " > " + targetFile.fsName);
445                 } else {
446                     this.showMsg("ファイル\n" + destFile.fsName + " : " + targetFile.fsName + "\nを確認しましょう。");
447                     this.pushLog(myAction + " : (error) " + destFile.fsName + " > " + targetFile.fsName);
448                 }
449                 break;
450             case    "deleteFile":
451                 var targetFile = new File(this.pathReplace(this.myInstall[idx][1]));
452 
453                 if (targetFile.exists) {
454                     if (this.debug) {
455                         alert(targetFile.fsName + " を削除した気分");
456                     } else {
457                         try {
458                             targetFile.remove();
459                         } catch (err) {
460                             this.abortProcess(myAction + ":\n" + err);
461                             return false;
462                         }
463                     }
464                     this.pushLog(myAction + " : " + targetFile.fsName);
465                 } else {
466                     alert(targetFile.fsName + " は無い様");
467                     this.pushLog(myAction + " : " + targetFile.fsName + "is not exists.");
468                 }
469                 break;
470             case    "doFunction":
471                 try {
472                     var myResult = (this.myInstall[idx][0](this.myInstall[idx][1]));
473                     if (!myResult) {
474                         return false;
475                     }
476                     this.pushLog("do userFunction");
477                 } catch (err) {
478                     this.abortProcess(myAction + ":\n" + err);
479                     return false;
480                 }
481                 break;
482             default:
483                 this.abortProcess(myAction + ":\n" + this.myInstall[idx].toString());
484                 return false;
485 
486         }
487     }
488     /**
489      * @desc アンインストールとインストールは、
490      * 実行するコマンド配列が異なるだけで同じ関数で処理される
491      */
492     return true;
