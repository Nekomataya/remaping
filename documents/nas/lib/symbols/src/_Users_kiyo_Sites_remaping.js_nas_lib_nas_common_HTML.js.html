 88             ui.item.option.selected = true;
 89             this._trigger( "select", event, {
 90               item: ui.item.option
 91             });
 92           },
 93  
 94           autocompletechange: "_removeIfInvalid"
 95         });
 96       },
 97

 99         var input = this.input,
100           wasOpen = false;
101  
102         $( "<a>" )
103           .attr( "tabIndex", -1 )
104           .attr( "title", "Show All Items" )
105           .tooltip()
106           .appendTo( this.wrapper )

110             },
111             text: false
112           })

133         response( this.element.children( "option" ).map(function() {
134           var text = $( this ).text();
135           if ( this.value && ( !request.term || matcher.test(text) ) )
