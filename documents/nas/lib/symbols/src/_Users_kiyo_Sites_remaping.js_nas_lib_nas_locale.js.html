 71 /**
 72  * nas.localize(anyString or languageResource [, replaceStrings ])
 73  * 引数:zStirng または　何らかの文字列　または　nas.LanguageResouce　第二引数以降は置換文字列
 74  * 内部で使用するlocalizeファンクションを作っておく
 75  * 置きかえ機能ありadobe拡張スクリプトのlocalizeの前段として使用可能nasオブジェクトの場合は自前で処理して
 76  * 引数が文字列で、かつZStringだった場合のみlocalizeのリザルトを戻す
 77  *
 78  * @param myObject
 79  * @returns {*}
 80  */
 81 nas.localize = function (myObject) {
 82     if ((typeof app != "undefined") && (localize instanceof Function) && (localize !== nas.localize ) && (myObject instanceof String) && (myObject.indexOf("$$$/") == 0)) {
 83         /**
 84          * 引数がZStringでかつAdobe環境(グローバルのlocalizeがある)なら渡して終了
 85          * localize  が自分自身である場合も排除
 86          * @type {string}
 87          */
 88         var myEx = "localize(myObject ";
 89         if (arguments.length > 1) {
 90             for (var aid = 1; aid < arguments.length; aid++) {
 91                 myEx += ",arguments[" + aid + "] "
 92             }
 93         }
 94         myEx += ");";
 95         return eval(myEx);
 96     } else {
 97         /**
 98          * 自前処理　localize がない場合(html等)は自前で処理
 99          * @type {Array}
100          */
101         var myArg = [];
102         if (arguments.length > 1) {
103             for (var aId = 1; aId < arguments.length; aId++) {
104                 myArg.push(arguments[aId]);
105             }
106         }
107         if (typeof myObject == "string") {
108             var myResult = myObject.replace(/^\$\$\$[^=]+=/, "");
109         } else {
110             var myResult = myObject[nas.locale];
111         }
112 
113         if (myArg.length > 0) {
114             /**
115              * 言語リソース内の%d(1,2,3…)を置き換える　　引数は不定
116              */
117             for (var rId = 0; rId < myArg.length; rId++) {
118                 var myRegex = new RegExp("%" + (rId + 1), "g");
119                 myResult = myResult.replace(myRegex, myArg[rId]);
120             }
121         }
122         return myResult;
123     }
124 };

153 /**
154  * @param myLocale
155  コマンドメニュー用の切り替えルーチンを追加
156  "_M"で開始するリソースは、pM/cM/ibMで兼用リソースとして扱われる
157  また"_M","innerHTML"に関しては接尾詞"-d"をカバー範囲にする
158  */
159 nas.LanguagePack.prototype.chgLocale = function (myLocale) {
160     if (!myLocale) {
161         myLocale = "ja"
162     }
163     for (var idx = 0; idx < nas.LangPack[myLocale].length; idx++) {
164         var eId = nas.LangPack[myLocale][idx][0];
165         var eType = nas.LangPack[myLocale][idx][1];
166         var value = nas.LangPack[myLocale][idx][2];
167         try {
168             if(eId.indexOf('pM')==0){
169                 var menuSet = ['pM','cM','ibM']
170                 for(pfx = 0 ; pfx < menuSet.length; pfx ++){
171                     if((pfx == 2)&&(eType=="innerHTML")) continue;
172                     var xeId = eId.replace(/^pM/,menuSet[pfx]);
173                     if (document.getElementById(xeId)) {
174                         document.getElementById(xeId)[eType] = value;
175                         if((eType=="innerHTML")&&(document.getElementById(xeId+'-d'))) {
176                             document.getElementById(xeId+'-d')['innerHTML'] = value;                        
177                         }
178                     }
179                 }
180             }else{
181                 if (document.getElementById(eId)) {
182                     document.getElementById(eId)[eType] = value;
183                 }
184             }
185         } catch (err) {
186             console.log(err);
187         }
188     }
189 };
