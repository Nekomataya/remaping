460     for (var myFrame = 0; myFrame < myXPS.duration(); myFrame++) {
461         currentRecord = [];
462         currentRecord.push((myFrame + 1).toString(10));//フレームカウント
463         for (var LC = 1; LC < referenceXPS.xpsTracks.length - 1; LC++) {
464             currentRecord.push('"' + referenceXPS.timeline(LC)[myFrame] + '"');//ダイアログフレームをとばす
465         }
466         for (var LC = 0; LC < myXPS.xpsTracks.length - 1 ; LC++) {
467             currentRecord.push('"' + sourceXPS.timeline(LC)[myFrame] + '"');//メモ情報はコンバートできない。
468         }
469         myStylosCSV.push(currentRecord.join(","));
470     }
471 
472     return myStylosCSV.join("\n");
473 }
