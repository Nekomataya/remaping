 45 /**
 46  * ベクトル演算事前処理与えられたベクトルの次数を多いものに揃えて不足分に0を加えて返す
 47  * @param vec1
 48  * @param vec2
 49  */
 50 function preformvector(vec1, vec2) {
 51     /**
 52      * 単項スカラだった場合、要素数1の配列に変換しておく。
 53      */
 54     if (typeof(vec1) == "number") {
 55         vec1 = [vec1];
 56     }
 57     if (typeof(vec2) == "number") {
 58         vec2 = [vec2];
 59     }
 60 
 61     /**
 62      * ベクトルの次数を求める 二次元か三次元か四次元か
 63      * @type {number}
 64      */
 65     var difD = (vec1.length - vec2.length);
 66     var vecD = (vec1.length > vec2.length) ? vec1.length : vec2.length;//多い方
 67 
 68     /**
 69      * 片方が不足する場合は0で補う
 70      */
 71     if (difD > 0) {
 72         for (var idx = 0; idx > difD; idx--) {
 73             vec2 = vec2.concat([0]);
 74         }
 75     }
 76     if (difD < 0) {
 77         for (var idx = 0; idx < difD; idx++) {
 78             vec1 = vec1.concat([0]);
 79         }
 80     }
 81     return [vec1, vec2, vecD];
 82 }

189  */
190 function clamp(vec, limit1, limit2) {
191     var max = limit1;
192     var min = limit2;
193     if (limit1 < limit2) {
194         max = limit2;
195         min = limit1;
196     }
197     if (typeof(vec) == "number") {
198         vec = [vec];
199     }
200     /**
201      * ベクトルの次数を求める 二次元か三次元か四次元か
202      *
203      */
204     var vecD = (vec.length);
205     var vecNew = new Array(vecD);
206     /**
207      * 要素ごとに値をクランプして返す。
208      *
209      */
210     for (idx = 0; idx < vecD; idx++) {
211         if (vec[idx] >= min && vec[idx] <= max) {
212             vecNew[idx] = vec[idx];
213         } else {
214             vecNew = (vec[idx] >= min ) ? vecNew.concat([max]) : vecNew = vecNew.concat([min]);
215         }
216     }
217     return vecNew;

546 /**
547  * AE互換?かもしれない?ここでは互換なし! t はフレーム数で与えること
548  * @param t
549  * @returns {*}
550  * @private
551  */
552 function valueAtTime_(t) {
553     if (t <= this[0].frame) {
554         return this[0].value
555     }
556     if (t >= this[this.length - 1].frame) {
557         return this[this.length - 1].value
558     }
559     for (id = 1; id < this.length; id++) {
560         if (t == this[id].frame) {
561             return this[id].value
562         }
563         /**
564          * 所属キーフレームが判明したので計算して返す
565          */
566         if (t < this[id].frame) {
567 
568             if (this[id].keyAtrib[0] == "stop") {
569                 //キー補完が停止の時は、補完計算なし。前方キーの値で返す。
570                 return this[id - 1].value;
571             } else {
572                 var Vstart = this[id - 1].value;
573                 var Vcp1 = this[id - 1].valueCp[0];
574                 var Vcp2 = this[id - 1].valueCp[1];
575                 var Vend = this[id].value;
576 
577                 var Tstart = this[id - 1].frame;
578                 var Tcp1 = this[id - 1].timingCp[0];
579                 var Tcp2 = this[id - 1].timingCp[1];
580                 var Tend = this[id].frame;
581 
582                 /**
583                  * 値が描くアークの全長を求める
584                  */
585                 var HallArk = nas.bezierL(Vstart, Vcp1, Vcp2, Vend);
586                 /**
587                  * 指定時間からタイミング係数を求める
588                  * @type {number}
589                  */
590                 var Now = (t - Tstart) / (Tend - Tstart);
591                 /**
592                  * 時間から 2次元(時間・比率)助変数を求める。
593                  */
594                 var Tvt = nas.bezierA(Tcp1[0], Tcp1[1], Now);
595                 /**
596                  * 求めた助変数でタイミング係数を出す
597                  */
598                 var Tvv = nas.bezierA(0, Tcp2[0], Tcp2[1], 1, Tvt);
599                 /**
600                  * 係数から値を求める。
601                  */
602                 Tt = Tvv;//仮助変数(初期値)
603                 Tmax = 1;
604                 Tmin = 0;
605                 var preLength = 0;//始点からのアーク長
606                 var postLength = 0;//終点までのアーク長
607                 var TtT = 0;//テストで得られる比率
608 
609                 do {
610                     preLength = nas.bezierL(Vstart, Vcp1, Vcp2, Vend, 0, Tt);
611                     postLength = nas.bezierL(Vstart, Vcp1, Vcp2, Vend, Tt, 1);
612                     TtT = preLength / (preLength + postLength);
613                     if (Tvv < preLength / (preLength + postLength)) {
614                         Tmin = Tt;//下限値を現在値に
615                         Tt = (Tmax + Tt) / 2;//新テスト値を設定
616                     } else {
617                         Tmax = Tt;//上限値を現在値に
618                         Tt = (Tmin + Tt) / 2;//新テスト値を設定
619                     }
620                 } while (TtT / Tvv > 0.9999999 && TtT / Tvv < 1.0000001);//精度確認
621                 //その得られた助変数を使って値を返す。値の次元数でループ
622                 var Result = new Array(Vstart.length);
623                 for (i = 0; i < Vstart.length; i++) {
624                     Result[i] = nas.bezier(Vstart[i], Vcp1[i], Vcp2[i], Vend[i], Tt)
625                 }

628         }
