 14     return myStr.replace(/[０１２３４５６７８９]/g,
 15         function (h) {
 16             var n = "０１２３４５６７８９".indexOf(h);

 18         }

 30 /**
 31  * @param TSHStream
 32  * @returns {*}
 33  * @constructor
 34  */
 35 function TSH2XPS(TSHStream) {
 36     /**
 37      * @desc データ冒頭のみチェックして明確に違うストリームの場合はエラーを返す
 38      */
 39     if (!TSHStream.match(/^\x22([^\x09]*\x09){25}[^\x09]*/)) {
 40         return false;
 41     }
 42     /**
 43      * CSVデータをオブジェクト化する
 44      * @type {{}}
 45      */
 46     myTSH = {};
 47 
 48     /**
 49      * ラインで分割して配列に取り込み
 50      * @type {Array}
 51      */
 52     myTSH.SrcData = [];
 53     if (TSHStream.match(/\r/)) {
 54         TSHStream = TSHStream.replace(/\r\n?/g, ("\n"));
 55     }
 56     TSHStream = TSHStream.replace(/\"/g, (""));//前後の\"を払う
 57     TSHStream = TSHStream.replace(/\n+/g, ("\n"));//空行があれば捨てる
 58     TSHStream = TSHStream.replace(/\n$/, "");//末尾が改行ならそれも捨てる
 59 
 60     myTSH.SrcData = TSHStream.split("\n");//改行で分割
 61 
 62     /**
 63      * @desc tshは
 64      * 第一レコードがセルラベル
 65      * 第二行は、キー変換の為のマーカー行だが、りまぴんには相当するプロパティが無いので無視
 66      * 第三レコード以降がシートデータ シートデータの最大有効レコードを取得して有効なレイヤ数を出すこと
 67      */
 68     myTSH.layerCount = 0;
 69     for (var lid = 2; lid < myTSH.SrcData.length; lid++) {
 70         if (myTSH.SrcData[lid].length > 25) {
 71             var currentCount = 26 - myTSH.SrcData[lid].match(/\t*$/)[0].length;
 72             //固定長レコードなので、全数から空フィールドの数をひいてCountをとる
 73             if (currentCount > myTSH.layerCount) {
 74                 myTSH.layerCount = currentCount
 75             }
 76         }
 77     }
 78     /**
 79      * 各ラインを更にテキストの配列に分解
 80      */
 81     for (var idx = 0; idx < myTSH.SrcData.length; idx++) {
 82         if (myTSH.SrcData[idx].length > 1) {
 83             myTSH.SrcData[idx] = myTSH.SrcData[idx].split("\t")
 84         }
 85     }
 86     myTSH.body = function (layerID, frameID) {
 87         return (this.SrcData[frameID + 2][layerID]);
 88     };
 89     /**
 90      * 前処理を済ませてフレーム継続数を取得 全有効ライン数-(ラベル行,マーク行)
 91      * @type {number}
 92      */
 93 
 94     myTSH.frameDuration = myTSH.SrcData.length - 2;
 95     /**
 96      * ラベル取得
 97      * @param layerID
 98      * @returns {*}
 99      */
100     myTSH.layerLabel = function (layerID) {
101         return myTSH.SrcData[0][layerID];
102     };
103 
104 
105     /**
106      * @desc シートから直接ラベルを取得するメソッド
107      */
108 
109     /**
110      * XPS互換ストリームに変換
111      * @returns {string}
112      */
113     myTSH.toSrcString = function () {
114 //	var myLineFeed=nas.GUI.LineFeed;
115         var myLineFeed = "\n";
116         var resultStream = "nasTIME-SHEET 0.4";
117         resultStream += myLineFeed;
118         resultStream += "#T-Sheet";
119         resultStream += myLineFeed;
120         resultStream += "##TIME=" + nas.Frm2FCT(this.frameDuration, 3, 0);
121         resultStream += myLineFeed;
122         resultStream += "##TRIN=0+00.,\x22\x22";
123         resultStream += myLineFeed;
124         resultStream += "##TROUT=0+00.,\x22\x22";
125         resultStream += myLineFeed;
126         /**
127          * ラベル配置
128          * @type {string}
129          */
130         resultStream += "[CELL\tN\t";
131         for (idx = 0; idx < this.layerCount; idx++) {
132             resultStream += this.layerLabel(idx) + "\t";
133         }
134         resultStream += "]";
135         resultStream += myLineFeed;
136 
137         for (frm = 0; frm < this.frameDuration; frm++) {
138             resultStream += "\t";
139             //T-Sheetはダイアログデータをサポートしないので1フィールドスキップ
140             resultStream += "\t";
141             for (idx = 0; idx < this.layerCount; idx++) {
142                 if (frm == 0) {
143                     var currentValue = this.body(idx, frm);
144                 } else {
145                     var currentValue = (this.body(idx, frm) == this.body(idx, (frm - 1))) ? "" : this.body(idx, frm);
146                 }
147                 resultStream += (currentValue === 0) ? "X\t" : currentValue + "\t";
148             }
149             resultStream += myLineFeed;
150         }
151         resultStream += "[END]";
152         resultStream += myLineFeed;
153         resultStream += "converted from T-Sheet";
154         return resultStream;
155     };
156     //alert( myTSH.toSrcString());
157     return myTSH.toSrcString();
158 }

160 /**
161  * 書式　XPS2TSH(myXPS)
162  * 引数はオブジェクトでも、ストリームでも受け付ける。
163  * コンバートするXPSをTSheet互換形式で書き出すことができる。
164  * 文字コードのコンバートは特にしていないので、
165  * 必要なら何か別のコンバート手段を利用してShift-JISに変換されたし。
166  *
167  * @param myXPS
168  * @returns {*}
169  * @constructor
170  */
171 function XPS2TSH(myXPS) {
172     /**
173      * 引数がソースであっても処理する。XPSでない場合はfalse
174      */
175     if (myXPS instanceof Xps) {
176         var sourceXPS = myXPS;
177     } else {
178         if ((myXPS instanceof String) && (myXPS.match(/^nasTIME-SHEET/))) {
179             var sourceXPS = new Xps();
180             if (!sourceXPS.readIN(myXPS)) {
181                 return false;
182             }
183         } else {
184             return false;
185         }
186     }
187 
188     /**
189      * XPSフォーマット拡張に従ってタイミング以外のデータが発生するので、種別判定して選択する機能が必要
190      * プロパティをチェックして必要なタイムラインのIDを抽出する
191      * @type {Array}
192      */
193     var myTarget = [];
194     for (var ix = 1; ix < myXPS.xpsTracks.length; ix++) {
195         if (myXPS.timeline(ix).option.match(/(replacement|timing|still)/i)) {
196             myTarget.push(ix);
197         }
198     }
199 
200     /**
201      * コンバートする
202      * @type {Array}
203      */
204     var myTSH = [];
205     myTSH.recordCount = 26;//TSHはレコード長固定
206     /**
207      * 第一レコードを作る
208      * @type {string}
209      */
210     var currentRecord = "";
211     var defaultNames = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
212     for (var ct = 0; ct < myTSH.recordCount; ct++) {
213         if (ct < myTarget.length) {
214             currentRecord += myXPS.timeline([myTarget[ct]]).id;
215         } else {
216             currentRecord += defaultNames.charAt(ct);
217         }
218         if (ct < (myTSH.recordCount - 1)) {
219             currentRecord += "\t";
220         }
221     }
222     myTSH.push(currentRecord);
223     /**
224      * 空レコードを挿入
225      */
226     myTSH.push(new Array(26).join("\t"));
227 
228     /**
229      * 第二レコード以降ボディデータを流し込む(桁揃えで固定長レコードにする)
230      */
231     for (var myFrame = 0; myFrame < myXPS.duration(); myFrame++) {
232         currentRecord = "";
233         for (var LC = 0; LC <= myTSH.recordCount; LC++) {
234 //            if (LC < myXPS.layers.length) {            }
235             if (LC < myTarget.length) {
236                 var currentValue = dataCheck( myXPS.xpsTracks[myTarget[LC]][myFrame], myXPS.timeline(myTarget[LC]).id);
237                 if (currentValue == "blank") {
238                     currentValue = "0";
239                 }
240                 if (currentValue == null) {
241                     currentValue = "";
242                 }
243                 currentRecord += currentValue + "\t";
244             } else {
245                 if (LC < (myTSH.recordCount - 1)) {
246                     currentRecord += "\t";
247                 }
248                 //空フィールド
249             }
250         }
251         myTSH.push(currentRecord);
252     }
253 
254     return '"' + myTSH.join("\r") + '"\r\n';
255 }
