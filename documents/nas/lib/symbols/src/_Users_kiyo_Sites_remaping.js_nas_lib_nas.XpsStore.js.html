1 ﻿/**
  2  * Ps等AE(乙女)以外の環境にXPSStoreを移植
  3  * コンポ参照を行なわず、Javascript環境全般に対応　要nas_lib
  4  *
  5  * @overview XPSStore　(シートトレーラ)
  6  *
  7  * XPSheetStore オブジェクト
  8  * プロジェクト内部にタイムシートを複数記録維持して切り換えて使用する仕組み
  9  * フッテージストア内部にコンポの形で記録する
 10  * インデックスがあり、選択値を切り換えてバッファとの間にやりとりを行う
 11  * バッファはXPS(=従来のXPSオブジェクトをそのまま使用)
 12  * XpsStore.body        実際にシートを格納する配列
 13  * 改行は"\\r\\n"に置換　内部的には各環境ごとのエンコーディングによる文字列
 14  * XpsStore.selected    カレントのタイムシートオブジェクト。カレントがない場合もあるその場合はnull または　false
 15  * デフォルトの値（ストア内のデータが無い）ではnull / ストア内にデータがありかつ選択の無い場合はfalseを値とする
 16  * AEではレイヤが配置されるが、他の環境ではXPSオブジェクトが配置されるので注意
 17  * XPS自体を取得する際はgetメソッドを使用すること
 18  * XPS.index プロパティが増設されている
 19  *
 20  * XpsStore.currentIndex    カレントインデックスを記録する内部変数　外部操作禁止
 21  *
 22  * XpsStore.getLength()    シート総数を返す
 23  * ０を返さずにfalseを返す場合は現状でシート操作環境がない事を表している。(この機能は乙女専用)
 24  * もし、シート操作環境が必要な場合は　XpsStore.initBody()メソッドで初期化すると環境が構築される。
 25  * XpsStore.get(index)    indexで指定されたレイヤからXpsオブジェクトを取得
 26  * XpsStore.set(index,Xps)    indexで指定されたレイヤにXpsオブジェクトを登録する
 27  * XpsStore.pop(index)    indexで指定されたシートのデータをXPSバッファに転送する。
 28  * indexが指定されなければカレントのシートに対して実行
 29  * XpsStore.push(index)    indexで指定されたシートにXPSバッファのデータをセットする。内部でsetInfo()を実行する。
 30  * indexが指定されなければカレントのシートに対して実行
 31  * XpsStore.select(index)    カレントインデックスを切り替えバッファ内容を更新する。
 32  * 戻り値は選択されているシートのインデックス。数値のほかにキーワード"fwd,prv,start,end"を受け付ける
 33  * XpsStore.getInfo(index)    指定indexのシートのモディファイド情報を取得
 34  * 戻り値は情報オブジェクト
 35  * indexが指定されなければカレントのシートに対して実行
 36  * XpsStore.setInfo(index)    指定indexのシートに現行のXPSバッファのモディファイド情報を設定
 37  * 戻り値は情報オブジェクト
 38  * indexが指定されなければカレントのシートに対して実行
 39  * XpsStore.add(Xps)    Xpsオブジェクトを直接渡して新規にシート(テキストレイヤ)を作成する。
 40  * XPSバッファを同時更新
 41  * カラ指定の場合はXPSバッファから作成
 42  * 既存シートとの重複は関知しないのであらかじめ確認は必要
 43  * 戻値は追加成功したテキストレイヤ
 44  * XpsStore.remove(index)    指定indexのシートをプロジェクトから削除する。インデックスは必ず指定すること。
 45  * インデックス不指定時は失敗させる
 46  * 成功時の戻り値は新たなバッファ内容(Xps)またはnull(残シートなしの時)
 47  * XpsStore.duplicate(index,name)    指定indexのシートをnameで複製する。インデックスは必ず指定すること。
 48  * 名前が指定されなかった場合は元のカット番号に"copy of "を前置する
 49  * 成功時の戻り値は新たなバッファ内容(Xps)　失敗時はfalse (未実装　2010 12 19)
 50  * XpsStore.setBody()    内部で使用するメソッド、ライブラリを参照してbodyを自身のオブジェクトとして再設定する
 51  * 設定済みの場合は何もしない。外部アクセス禁止
 52  *
 53  * XpsStore.initBody()    内部で使用する初期化メソッド
 54  カラのトレーラを作るので注意
 55  */

469     /**
470      * 関連付けられているコンポのupdate//このメソッドはAEのみで意味を持つので他の環境ではコールしないこと
471      * @param indx
472      * @returns {*}

474     this.update = function (indx) {
475         return false;
476         /**
477          * indxはシートindex =シートキャリアのレイヤインデックスと一致　1〜
478          */
479         var myIdentifier = this.get(indx).getIdentifier('cut');
480         /**
481          * チェックするシートの識別を取得
482          * アイテム総当り
483          * @type {number}
484          */
485         var myResult = 0;
486         for (var idx = 1; idx <= app.project.items.length; idx++) {
487 
488             /**
489              * CompItem && ステージプレフィックスありコメントシグネチャあり
490              * 3条件を自動更新の必要条件にする…かためカモ
491              */
492             if (
493                 (app.project.item(idx) instanceof CompItem) &&
494                 (app.project.item(idx).name.match(new RegExp("^\\(" + nas.itmFootStamps.stage[0] + "\\)"))) &&
495                 (app.project.item(idx).comment.match(new RegExp("^" + nas.itmFootStamps.stage[1])))
496             ) {
497                 var myXps = app.project.item(idx).getRootXps();//コンポからXPS取得
498                 if (myXps) {
499                     /**
500                      * Xpsがあれば識別情報を比較　一致したら適用
501                      */
502                     if (myIdentifier == myXps.getIdentifier('cut')) {
503                         app.project.item(idx).applyXPS(myXps);
504                         nas.otome.writeConsole("updated Comp [" + idx + "] " + app.project.item(idx).name);
505                         //	myComps.push(app.project.item(idx));
506                         myResult++;
507                     }
508                 }
509             }
510         }
511         return myResult;
