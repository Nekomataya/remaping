 31     /**
 32      * @param myLayer
 33      * @returns {*|Window|SymbolInstance|string|null}
 34      */
 35     nas.getRootDocument = function (myLayer) {
 36         var result = myLayer.parent;
 37         while (result.typename != "Document") {
 38             result = result.parent;
 39         }
 40         return result;
 41     };
 42     /**
 43      * @param myLayer
 44      */
 45     nas.setActive = function (myLayer) {
 46         app.activeDocument = nas.getRootDocument(myLayer);
 47         app.activeDocument.activeLayer = myLayer;
 48     };
 49 
 50     /**
 51      * @param layer
 52      * @param pt
 53      */
 54     nas.PSCCFontSizeFix.setFontSizePoints = function (layer, pt) {
 55         nas.setActive(layer);
 56         function cTID(s) {
 57             return app.charIDToTypeID(s);
 58         }
 59 
 60         function sTID(s) {
 61             return app.stringIDToTypeID(s);
 62         }
 63 
 64         var magicNumber = cTID("0042");
 65 
 66         var desc23 = new ActionDescriptor();
 67 
 68         var ref6 = new ActionReference();
 69         ref6.putProperty(cTID('Prpr'), cTID('TxtS'));
 70         ref6.putEnumerated(cTID('TxLr'), cTID('Ordn'), cTID('Trgt'));
 71         desc23.putReference(cTID('null'), ref6);
 72 
 73         var desc24 = new ActionDescriptor();
 74         desc24.putInteger(sTID('textOverrideFeatureName'), magicNumber);
 75         desc24.putInteger(sTID('typeStyleOperationType'), 3);
 76         desc24.putUnitDouble(cTID('Sz  '), cTID('#Pnt'), pt);
 77         desc23.putObject(cTID('T   '), cTID('TxtS'), desc24);
 78 
 79         executeAction(cTID('setd'), desc23, DialogModes.NO);
 80 
 81     };
 82

117     nas.PSCCFontSizeFix.test = function () {
118         var doc = app.documents.add(UnitValue("5 in"), UnitValue("7 in"), 300);
119         var layer = doc.artLayers.add();
120         layer.kind = LayerKind.TEXT;
121         layer.name = "Test";
122         var titem = layer.textItem;
123 
124         titem.size = new UnitValue("50", "pt");
125 
126         if (Math.round(titem.size.as("pt")) != 50) {
127             nas.PSCCFontSizeFix.setFontSizePoints(layer, 50);
128         }
129 
130         titem.contents = "This text should be 50pt";
131         titem.font = "Monaco";
132         alert(Math.round(titem.size.as("pt")) + " pt");
133 
134         doc.close(SaveOptions.DONOTSAVECHANGES);
135 
136         var doc = app.documents.add(UnitValue("5 in"), UnitValue("7 in"), 300);
137         var layer = doc.artLayers.add();
138         layer.kind = LayerKind.TEXT;
139         layer.name = "Test";
140         var titem = layer.textItem;
141         titem.size = new UnitValue("50", "px");
142 
143         if (Math.round(titem.size.as("px")) != 50) {
144             nas.PSCCFontSizeFix.setFontSizePixels(layer, 50);
145         }
146 
147         titem.contents = "This text should be 50px/12pt";
148         titem.font = "Monaco";
149         alert(Math.round(titem.size.as("px")) + " px");
150 
151         //  doc.close(SaveOptions.DONOTSAVECHANGES);
152     };
