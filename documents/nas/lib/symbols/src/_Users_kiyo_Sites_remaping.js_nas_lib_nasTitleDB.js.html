 83 /**
 84  * コンストラクタ
 85  * usage: new inputMedia([width[,aspect[,fps[,resolution[,frame]]]]])
 86  *
 87  * @constructor
 88  * @param width
 89  * @param aspect
 90  * @param fps
 91  * @param resolution
 92  * @param frame
 93  * @param peg
 94  */
 95 function inputMedia(width, aspect, fps, resolution, frame, peg) {
 96     this.frameWidth = (!isNaN(width)) ? width : 24.50;//cm
 97     this.frameAspect = (!isNaN(aspect)) ? aspect : (16 / 9);//width/height
 98     this.frameRate = (!isNaN(fps)) ? fps : 24;//fps
 99     this.paintResolution = (!isNaN(resolution)) ? resolution : 144 / 2.54;//dpc
100     this.baseFrame = (!isNaN(frame)) ? frame : 100;//撮影フレーム
101     this.pegbarOffset = (!isNaN(peg)) ?
102         peg : [0, 3.0 + (.5 * this.frameWidth / this.frameAspect), 0];
103     // 標準タップ位置([ローカル座標X(cm),ローカル座標Y(cm),ローテーション(度)])
104 }
105 /**
106  * usage: new outputMedia([width[,aspect[,fps[,resolution[,frame]]]]])
107  *
108  * @param lines
109  * @param width
110  * @param aspect
111  * @param fps
112  * @param interlace
113  */
114 function outputMedia(lines, width, aspect, fps, interlace) {
115     this.lines = (!isNaN(lines)) ? lines : 480;//lines/frame
116     this.width = (!isNaN(width)) ? width : 640;//pix/line
117     this.pixAspext = (!isNaN(aspect)) ? aspect : 1;//width/height
118     this.frameRate = (!isNaN(fps)) ? fps : 24;//fps
119     this.isInterlace = (interlace) ? true : false;//interlace
120 }
