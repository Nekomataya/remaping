345 /**
346  * リスト展開プロシージャ
347  *
348  * nas_expdList(ListStr,rcl)
349  * 引数:    ソース文字列　ListStr /  rcl
350  * 戻値:    putメソッドの引数ストリーム
351  * マクロ記法の文字列をputメソッドに引き渡し可能なストリームへ展開する
352  * リスト展開エンジンは汎用性を持たせたいので、無理やりグローバルに置いてある。
353  * 要注意
354  */
355 
356 var expd_repFlag = false;
357 var expd_skipValue = 0;//グローバルで宣言
358 
359 /**
360  * リスト展開はxUIのメソッドか?
361  * @param ListStr
362  * @param rcl
363  * @returns {*}
364  */
365 function nas_expdList(ListStr, rcl) {
366     if (typeof rcl == "undefined") {
367         rcl = false
368     } else {
369         rcl = true
370     }
371     var leastCount = (xUI.Selection[1]) ? xUI.Selection[1] : XPS.duration() - xUI.Select[1];
372     if (!rcl) {
373         expd_repFlag = false;
374         expd_skipValue = xUI.spinValue - 1;
375         //再帰呼び出し以外はスピン値で初期化
376     }
377     /**
378      * (スキップ量はスピン-１)この値はグローバルの値を参照
379      * @type {string}
380      */
381     var SepChar = "\.";
382 //	var SepChars=["\.","\ ","\,"];
383 
384     /**
385      * 台詞レイヤの場合のみ、カギ括弧の中をすべてセパレートする
386      * ダイアログトラックは固定ではなくなったので判定を変更
387      *  コメントトラックを排除する必要あり
388      * この判定をxUIに依存すると汎用性がなくなるので、コール側で引数渡しに変更する必要あり？
389      */
390     if (
391         (xUI.Select[0] == 0) ||
392         ((xUI.Select[0] < (XPS.xpsBody.length - 2)) &&
393         (XPS.layers[xUI.Select[0] - 1].option == "dialog"))
394     ) {
395         if (ListStr.match(/「([^「]*)」?/)) ;
396         if (ListStr.match(/「(.+)」?/)) {
397 //alert("Hit ："+ListStr.match(/^(.*「)([^」]*)(」?$)/));
398             ListStr = d_break(ListStr.match(/^(.*「)([^」]*)(」?$)/));
399             ListStr = ListStr.replace(/「/g, SepChar + "「" + SepChar);//開き括弧はセパレーション
400         }
401         ListStr = ListStr.replace(/\」/g, "---");//閉じ括弧は横棒
402         ListStr = ListStr.replace(/\、/g, "・");//読点中黒
403         ListStr = ListStr.replace(/\。/g, "");//句点空白(null)
404         ListStr = ListStr.replace(/\ー/g, "｜");//音引き縦棒
405     }
406     /**
407      * r導入リピートならば専用展開プロシージャへ渡してしまう
408      */
409     if (ListStr.match(/^([\+rR])(.*)$/)) {
410         var expdList = TSX_expdList(ListStr);
411         expd_repFlag = true;
412     } else {
413 
414         /**
415          * リスト文字列を走査してセパレータを置換
416          * @type {void|XML|string}
417          */
418         ListStr = ListStr.replace(/[\,\x20]/g, SepChar);
419         /**
420          * スラッシュを一組で括弧と置換(代用括弧)
421          * @type {XML|string}
422          */
423         ListStr = ListStr.replace(/\/(.*)(\/)/g, "\($1\)");//コメント引数注意
424 //		var PreX="/\(\.([1-9])/g";//括弧の前にセパレータを補う
425         ListStr = ListStr.replace(/\(([^\.])/g, "\(\.$1");
426 //		var PostX="/[0-9](\)[1-9])/";//括弧の後にセパレータを補う
427         ListStr = ListStr.replace(/([^\.])(\)[1-9]?)/g, "$1\.$2");
428 
429         /**
430          * 前処理終わり
431          * リストをセパレータで分割して配列に
432          */
433         var srcList = [];
434         srcList = ListStr.toString().split(SepChar);
435 
436 //	if (xUI.Select[0]==0){
437 //	} else {}
438 
439         var expdList = [];//生成データ配列を作成
440         var sDepth = 0;//括弧展開深度/初期値0
441         var StartCt = 0;
442         var EndCt = 0;
443 //rT=0;
444         /**
445          * 元配列を走査
446          * @type {number}
447          */
448         var ct = 0;//ローカルスコープにするために宣言する
449         for (ct = 0; ct < srcList.length; ct++) {
450             tcn = srcList[ct];
451 
452             /**
453              * トークンが開きカギ括弧の場合リザルトに積まないで
454              * リザルトのおしまいの要素を横棒にする。
455              */
456 //	if (tcn.match(/^[「」]$/)) {y_bar();continue;}
457             if (tcn == "「") {
458                 y_bar();
459                 continue;
460             }
461 
462             /**
463              * トークンがコントロールワードならば値はリザルトに積まない
464              * 変数に展開してループ再開
465              */
466             if (tcn.match(/^s([1-9][0-9]*)$/)) {
467 //		if(RegExp.$1*1>0) {xUI.spin(RegExp.$1*1)}else{xUI.spin(1)};
468                 expd_skipValue = (RegExp.$1 * 1 > 0) ? (RegExp.$1 * 1 - 1) : 0;
469                 continue;
470             }
471             /**
472              * グローバル
473              * トークンが開き括弧ならばデプスを加算して保留
474              */
475             if (tcn.match(/^(\(|\/)$/)) {
476                 sDepth = 1;
477                 StartCt = ct;
478                 /**
479                  * トークンを積まないで閉じ括弧を走査
480                  * @type {number}
481                  */
482                 var ct2 = 0;//ローカルスコープにするために宣言する
483                 for (ct2 = ct + 1; ct2 < srcList.length; ct2++) {
484                     if (srcList[ct2].match(/^\($/)) {
485                         sDepth++
486                     }
487                     if (srcList[ct2].match(/^(\)|\/)[\*x]?([0-9]*)$/)) {
488                         sDepth--
489                     }
490                     if (sDepth == 0) {
491                         EndCt = ct2;
492                         /**
493                          * 最初の括弧が閉じたので括弧の繰り返し分を取得/ループ
494                          * @type {number}
495                          */
496                         var rT = RegExp.$2 * 1;
497                         if (rT < 1) {
498                             rT = 1
499                         }
500                         if (RegExp.$2 == "") {
501                             expd_repFlag = true;
502                         }
503                         var ct3 = 0;//ローカルスコープにするために宣言する
504                         for (ct3 = 1; ct3 <= rT; ct3++) {
505                             if ((StartCt + 1) != EndCt) {
506 //alert("DPS= "+sDepth+" :start= "+StartCt+"  ;end= "+EndCt +"\n"+ srcList.slice(StartCt+1,EndCt).join(SepChar)+"\n\n-- "+rT);
507                                 expdList = expdList.concat(nas_expdList(srcList.slice(StartCt + 1, EndCt).join(SepChar), "Rcall"));
508                                 /**
509                                  * 括弧の中身を自分自身に渡して展開させる
510                                  * 展開配列が規定処理範囲を超過していたら処理終了
511                                  */
512                                 if (expdList.length >= leastCount) {
513                                     return expdList
514                                 }
515                             }
516                         }
517                         ct = EndCt;
518                         break;
519                     }//if block end
520                 }//ct2 loop end
521                 if (rT == 0) {
522                     expdList.push(srcList[ct]);
523                     s_kip();//ct++;
524                 }
525             } else {
526                 /**
527                  * トークンが展開可能なら展開して生成データに積む
528                  */
529                 if (tcn.match(/^([1-9]{1}[0-9]*)\-([1-9]{1}[0-9]*)$/)) {
530                     var stV = Math.round(RegExp.$1 * 1);
531                     var edV = Math.round(RegExp.$2 * 1);
532                     if (stV <= edV) {
533                         for (tcv = stV; tcv <= edV; tcv++) {
534                             expdList.push(tcv);
535                             s_kip();
536                         }
537                     } else {
538                         for (tcv = stV; tcv >= edV; tcv--) {
539                             expdList.push(tcv);
540                             s_kip();
541                         }
542                     }
543                 } else {
544                     expdList.push(tcn);
545                     s_kip();
546                 }
547 
548             }
549         }
550     }
551     /**
552      * 生成配列にスキップを挿入
553      */
554     function s_kip() {
555 //for (x=0;x<(xUI.spinValue-1);x++){expdList.push('');}
556         for (x = 0; x < expd_skipValue; x++) {
557             expdList.push('');
558         }
559     }
560 
561     /**
562      * 配列の末尾を横棒に
563      */
564     function y_bar() {
565         expdList.pop();
566         expdList.push('---');
567     }
568 
569     /**
570      * かぎ括弧の中身をセパレーション
571      * @param dList
572      * @returns {*}
573      */
574     function d_break(dList) {
575         wLists = dList.toString().split(",");
576         return wLists[1] + wLists[2].replace(/(.)/g, "$1\.") + wLists[3];
577     }
578 
579 
580     /**
581      * カエス
582      */
583     if (expdList.length < leastCount && expd_repFlag) {
584         blockCount = expdList.length;
585 //			alert(blockCount + " / " +leastCount);
586         for (resultCt = 0; resultCt <= (leastCount - blockCount); resultCt++) {
587             expdList.push(expdList[resultCt % blockCount]);
588         }
589     }
590     return expdList.join();//文字列で戻す
591 //		return expdList;
592 }

795 /**
796  * テンプレートを利用したeps出力
797  * テンプレートは、サーバ側で管理したほうが良いのだけど　一考
798  * XPSから出力に必要な本体データを切り出し、1ページづつepsエンコードして返す
799  *
800  * 引数は整数　ページナンバー 1から開始
801  * 引数が0 又は引数なしは全ページリザルト
802  * ページが存在しない場合は空データを返す
803  *
804  * @param myPage
805  * @returns {*}
806  */
807 getBodyData = function (myPage) {
808 
809     var startCount = 0;
810     var endCount = XPS.duration();
811     if ((myPage > 0 ) && (myPage <= Math.ceil(XPS.duration() / xUI.PageLength))) {
812         startCount = (myPage - 1) * xUI.PageLength;
813         endCount = (endCount > (startCount + xUI.PageLength)) ? startCount + xUI.PageLength : endCount;
814     } else {
815         if (myPage > Math.ceil(XPS.duration() / xUI.PageLength))return "";
816     }
817     var myBody = [];
818     for (var frm = startCount; frm < endCount; frm++) {
819         for (var col = 0; col < (XPS.xpsBody.length); col++) {
820             var currentData = XPS.xpsBody[col][frm];
821             if (currentData.match(/^[|｜:]$/)) {
822                 currentData = ""
823             }
824             myBody.push("\(" + EncodePS2(currentData) + ")");
825         }
826     }
827     return myBody.join(" ");
828 };
