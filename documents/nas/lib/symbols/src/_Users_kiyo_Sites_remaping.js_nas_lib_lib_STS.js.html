 23 /**
 24  * STSファイルをXPS互換テキストにコンバートする
 25  *
 26  * @param {Object File} myOpenFile
 27  *  ESTKのファイルオブジェクトを直接渡す
 28  * @returns {String Xps}
 29  * Xps互換テキストストリームを返す
 30  * @constructor
 31  */
 32 function STS2XPS(myOpenFile) {
 33     /**
 34      * 識別文字列位置を確認してファイルフォーマット判定
 35      */
 36     myOpenFile.open("r");
 37     checkVer = myOpenFile.read(18);
 38     myOpenFile.close();
 39     if (!checkVer.match(/^\x11ShiraheiTimeSheet$/)) {
 40         return false;
 41     }
 42 
 43 
 44     /**
 45      * オープンして配列にとる
 46      */
 47     myOpenFile.open("r");
 48     mySTS = myOpenFile.getBin();
 49     myOpenFile.close();
 50     /**
 51      * STSデータをオブジェクト化する
 52      */
 53     mySTS.file = myOpenFile;//参照可能なようにオリジナルのファイルをプロパティｌにぶら下げる
 54     mySTS.frameDuration = mySTS[19] * 1 + mySTS[20] * 256;
 55     //フレーム継続数
 56     mySTS.layerCount = mySTS[18] * 1;
 57     //レイヤ数
 58     mySTS.dataLength = 2;
 59     //1フレームあたりのデータ長2バイト整数
 60     mySTS.body = function (layerID, frameID) {
 61         //2bite/1data : offset 23bite : IDは0オリジン アドレスを計算して値を戻すメソッド
 62         var myAddress = (layerID) * (this.frameDuration * this.dataLength) + (frameID * this.dataLength) + 23;
 63         return this[myAddress] + this[myAddress + 1] * 256;
 64     };
 65 
 66     /**
 67      * ラベル取得はファイル読み込みで(S-JIS)*大丈夫かな?
 68      * @type {Array}
 69      */
 70     mySTS.layerLabel = new Array(mySTS.layerCount);//ラベル配列
 71     var labelDataLength = new Array(mySTS.layerCount);//ラベルデータ長配列
 72 
 73     /**
 74      * ラベルの位置と長さを取得
 75      */
 76 
 77     /**
 78      * ラベル0のシーク位置
 79      * @type {number}
 80      */
 81     var labelOffset = mySTS.layerCount * (mySTS.frameDuration * mySTS.dataLength) + 23;
 82     /**
 83      * ラベル長(バイト数)
 84      */
 85     labelDataLength[0] = mySTS[labelOffset];
 86     /**
 87      * open
 88      */
 89     myOpenFile.open("r");
 90     myOpenFile.encoding = "CP932";
 91     /**
 92      * 最初のラベルを取得
 93      */
 94     myOpenFile.seek(labelOffset + 1, 0);
 95     mySTS.layerLabel[0] = myOpenFile.read(labelDataLength[0]);
 96     for (idx = 1; idx < mySTS.layerCount; idx++) {
 97         labelOffset = labelOffset + mySTS.layerLabel[idx - 1].length + 1;//新アドレス
 98         labelDataLength[idx] = mySTS[labelOffset];//ラベル長(バイト数)
 99         myOpenFile.seek(labelOffset + 1, 0);//シーク
100         mySTS.layerLabel[idx] = myOpenFile.read(labelDataLength[idx]);//取得
101     }
102     /**
103      * close
104      */
105     myOpenFile.close();
106     /*
107      * XPS互換ストリームに変換
108      * @returns {string}
109      */
110     mySTS.toSrcString = function () {
111         var resultStream = "nasTIME-SHEET 0.4";
112         resultStream += nas.GUI.LineFeed;
113         resultStream += "#ShiraheiTimeSheet";
114         resultStream += nas.GUI.LineFeed;
115         resultStream += "##TITLE=" + nas.workTitles.selectedName;
116         resultStream += nas.GUI.LineFeed;
117         resultStream += "##CUT=" + this.file.name.replace(/\.[^\.]+$/, "");
118         resultStream += nas.GUI.LineFeed;
119         resultStream += "##TIME=" + nas.Frm2FCT(this.frameDuration, 3, 0);
120         resultStream += nas.GUI.LineFeed;
121         resultStream += "##TRIN=0+00.,\x22\x22";
122         resultStream += nas.GUI.LineFeed;
123         resultStream += "##TROUT=0+00.,\x22\x22";
124         resultStream += nas.GUI.LineFeed;
125         /**
126          * ラベル配置
127          * @type {string}
128          */
129         resultStream += "[CELL\tN\t";
130         for (idx = 0; idx < this.layerCount; idx++) {
131             resultStream += this.layerLabel[idx] + "\t";
132         }
133         resultStream += "]";
134         resultStream += nas.GUI.LineFeed;
135 
136         for (frm = 0; frm < this.frameDuration; frm++) {
137             resultStream += "\t\t";
138             for (idx = 0; idx < this.layerCount; idx++) {
139                 if (frm == 0) {
140                     var currentValue = this.body(idx, frm);
141                 } else {
142                     var currentValue = (this.body(idx, frm) == this.body(idx, (frm - 1))) ? "-" : this.body(idx, frm);
143                 }
144                 resultStream += (currentValue === 0) ? "X\t" : currentValue + "\t";
145             }
146             resultStream += nas.GUI.LineFeed;
147         }
148         resultStream += "[END]";
149         resultStream += nas.GUI.LineFeed;
150         resultStream += "Convert from STS";
151         return resultStream;
152     };
153     return mySTS.toSrcString();
154 }
