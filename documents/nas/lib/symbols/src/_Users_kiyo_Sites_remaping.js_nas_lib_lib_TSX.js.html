 37 /**
 38  * TSX形式の繰り返し処理を配列に展開して返す
 39  * @function
 40  * @param {String} ListStr
 41  *  TSX形式繰り返し記述
 42  * @param rcl
 43  *  再帰呼出しフラグ
 44  * @returns {Array}
 45  *  セル番号の配列
 46  */
 47 function TSX_expdList(ListStr, rcl) {
 48     if (!rcl) {
 49         rcl = false
 50     } else {
 51         rcl = true
 52     }
 53     if (!rcl) {
 54         TSXEx_skipValue = 0;
 55         //初回呼び出しのみ再度初期化
 56     }
 57     /*
 58      * (スキップ量は0固定)
 59      */
 60     var SepChar = "\,";//カンマのみ？
 61 
 62     /*
 63      * 冒頭の[rR+]を削除
 64      */
 65     ListStr = ListStr.replace(/^([\+rR])(.*)$/, "$2");
 66     /*
 67      * リスト文字列を走査してカラセル表記を置換
 68      */
 69     ListStr = ListStr.replace(/[X\* ]/g, "x");
 70     /*
 71      * 角括弧を一組で括弧と置換
 72      */
 73     ListStr = ListStr.replace(/\[(.*)\]/g, "\($1\)");
 74     /*
 75      * "="を展開
 76      */
 77     ListStr = ListStr.replace(/(\,?)(\(.+\))\=([1-9][0-9]*)/g, "$1s$3\,$2\,s1");//括弧あり
 78     ListStr = ListStr.replace(/([\,^]+)([1-9][0-9]*)\=([1-9][0-9]*)/g, "$1s$3\,$2\,s1");//括弧なし
 79 //		var PreX="/\(\.([1-9])/g";//括弧の前にセパレータを補う
 80     ListStr = ListStr.replace(/\(([^\.])/g, "\(" + SepChar + "$1");
 81 //		var PostX="/[0-9](\)[1-9])/";//括弧の後にセパレータを補う
 82     ListStr = ListStr.replace(/([^\.])(\)[1-9]?)/g, "$1" + SepChar + "$2");
 83 
 84     /*
 85      * 前処理終わり
 86      * リストをセパレータで分割して配列に
 87      */
 88     var srcList = [];
 89     srcList = ListStr.toString().split(SepChar);
 90 
 91     if (xUI.Select[0] == 0) {
 92     } else {
 93     }
 94 
 95     var expdList = [];//生成データ配列を作成
 96     var sDepth = 0;//括弧展開深度/初期値0
 97     var StartCt = 0;
 98     var EndCt = 0;
 99 //rT=0;
100     /*
101      * 元配列を走査
102      */
103     for (var ct = 0; ct < srcList.length; ct++) {
104         tcn = srcList[ct];
105 
106         /*
107          * トークンがコントロールワードならば値はリザルトに積まない
108          * 変数に展開してループ再開
109          */
110         if (tcn.match(/^s([1-9][0-9]*)$/)) {
111 //		if(RegExp.$1*1>0) {xUI.spin(RegExp.$1*1)}else{xUI.spin(1)};
112             TSXEx_skipValue = (RegExp.$1 * 1 > 0) ? (RegExp.$1 * 1 - 1) : 0;
113             continue;
114         }
115 
116         /*
117          * グローバル
118          * トークンが開き括弧ならばデプスを加算して保留
119          */
120         if (tcn.match(/^(\(|\/)$/)) {
121             sDepth = 1;
122             StartCt = ct;
123             /*
124              * トークンを積まないで閉じ括弧を走査
125              */
126             var ct2 = 0;//ローカルスコープにするために宣言する
127             for (ct2 = ct + 1; ct2 < srcList.length; ct2++) {
128                 if (srcList[ct2].match(/^\($/)) {
129                     sDepth++
130                 }
131                 if (srcList[ct2].match(/^(\)|\/)[\*x]?([0-9]*)$/)) {
132                     sDepth--
133                 }
134                 if (sDepth == 0) {
135                     EndCt = ct2;
136                     /*
137                      * 最初の括弧が閉じたので括弧の繰り返し分を取得/ループ
138                      */
139                     var rT = RegExp.$2 * 1;
140                     if (rT < 1) {
141                         rT = 1
142                     }
143                     var ct3 = 0;//ローカルスコープにするために宣言する
144                     for (ct3 = 1; ct3 <= rT; ct3++) {
145                         if ((StartCt + 1) != EndCt) {
146 //alert("DPS= "+sDepth+" :start= "+StartCt+"  ;end= "+EndCt +"\r\n"+ srcList.slice(StartCt+1,EndCt).join(SepChar)+"\r\n\r\n-- "+rT);
147                             expdList = expdList.concat(TSX_expdList(srcList.slice(StartCt + 1, EndCt).join(SepChar), "Rcall"));
148                             /*
149                              * 括弧の中身を自分自身に渡して展開させる
150                              */
151                         }
152                     }
153                     ct = EndCt;
154                     break;
155                 }//if block end
156             }//ct2 loop end
157             if (rT == 0) {
158                 expdList.push(srcList[ct]);
159                 tss_kip();//ct++;
160             }
161         } else {
162             /*
163              * トークンが展開可能なら展開して生成データに積む
164              */
165             if (tcn.match(/^([1-9]{1}[0-9]*)\-([1-9]{1}[0-9]*)$/)) {
166                 var stV = Math.round(RegExp.$1 * 1);
167                 var edV = Math.round(RegExp.$2 * 1);
168                 if (stV <= edV) {
169                     for (tcv = stV; tcv <= edV; tcv++) {
170                         expdList.push(tcv);
171                         tss_kip()
172                     }
173                 } else {
174                     for (tcv = stV; tcv >= edV; tcv--) {
175                         expdList.push(tcv);
176                         tss_kip()
177                     }
178                 }
179             } else {
180                 expdList.push(tcn);
181                 tss_kip();
182             }
183 
184         }
185     }
186     /*
187      * 生成配列にスキップを挿入
188      */
189     function tss_kip() {
190         for (x = 0; x < TSXEx_skipValue; x++) {
191             expdList.push('');
192         }
193     }
194     /*
195      * カエス
196      */
197     return expdList;
198 }

200 /**　XpsからTSXテキストへの変換と書き出し
201  * @param {Object Xps} obj
202  *  変換対象のXps
203  * @param layerID
204  *  書き出し対象のレイヤID(トラックID+1)
205  */
206 function writeTSX(obj, layerID) {
207     if (!isNaN(layerID)) {
208         var stID = layerID;
209         var edID = layerID + 1;
210     } else {
211         var stID = 1;
212         var edID = obj.xpsTracks.length-1;
213     }
214     /*
215      * objはXPSオブジェクトを与えること
216      */
217 //対象タイムラインがtiming系でない場合は、処理をスキップ
218     for (column = stID; column < edID; column++) {
219         if (!obj.xpsTracks[column].option.match(/(replacement|timing|still)/i)) {
220             continue
221         }
222 //for (LID=stID;LID<edID;LID++){}
223         result = "";//リザルト初期化
224 //	column=LID
225 
226         /**
227          * シートボディ
228          */
229         for (line = 0; line < obj.duration(); line++) {
230             dCk = dataCheck(obj.xpsTracks[column][line], obj.xpsTracks[column].id, true);
231             switch (dCk) {
232                 case    null    :
233                     result += "";
234                     break;
235                 case    "blank":
236                     result += "*";
237                     break;
238                 default:
239                     result += dCk;
240             }
241 //		if((column+1)!=obj.layers.length){result+='\t'};//セパレータ
242             result += '--' + (line + 1).toString();//行番号
243             if (line == 0) {
244                 result += "\t\[ " + obj.xpsTracks[column].id + " \]"
245             }
246             //レイヤ名
247             result += '\r\n';//改行
248         }
249         /**
250          * ENDマーク
251          * @type {string}
252          */
253         result += 'E';
254 
255 // // // // //返す(とりあえず)
256 //return result;
257         wiOpt = "screenX=";
258         wiOpt += (column * 32).toString();
259         wiOpt += ",screenY=";
260         wiOpt += column * 24;
261         wiOpt += ",left=";
262         wiOpt += column * 32;
263         wiOpt += ",top=";
264         wiOpt += column * 24;
265         wiOpt += ",width=480,height=360,scrollbars=yes,menubar=yes";
266 
267         _w = window.open("", obj.xpsTracks[column].id, wiOpt);
268 
269         _w.document.open("text/plain");
270         if (!MSIE && !Firefox)_w.document.write("<html><body><pre>");
271         _w.document.write(result);
272         if (!MSIE && !Firefox)_w.document.write("</pre></body></html>");
273         _w.document.close();
274         _w.window.document.title = obj.xpsTracks[column].id.toString();
275     }
276 //	return false;//リンクのアクションを抑制するためにfalseを返す

280 /**
281  * TSXストリームをXPS互換データストリームに変換する
282  * ストリームが、複数のレイヤを含んでいても良い
283  * 
284  * @function
285  * @param {String} TSX datastream
286  * @returns {String XpsStream}
287  *
288  */
289 function TSX2XPS(datastream) {
290     /*
291      * データ冒頭の空白文字を削除
292      */
293     datastream = datastream.replace(/^\s*/, "");
294     if ((!datastream.toString().length ) ||
295         ( false )
296     ) {
297         alert("error : " + datastream);
298         return false
299     }
300     /*
301      * 不正データ時処理
302      * ラインで分割して配列に取り込み
303      */
304     if (datastream.match(/\r/)) {
305         datastream = datastream.replace(/\r\n?/g, ("\n"));
306     }
307     var SrcData = [];
308     SrcData = datastream.split("\n");
309 
310     /*
311      * データストリームを判定する
312      */
313     SrcData.startLine = 0;//データ開始行
314     /*
315      * ソースデータのプロパティ
316      */
317     SrcData.layerCount = 0;//レイヤ数
318     SrcData.layers     = [];//レイヤ情報トレーラー
319     SrcData.frameCount = 0;//読み取りフレーム数
320     SrcData.time       = 0;//初期化
321     var LayerDuration  = 0;
322     var LayerCount     = 0;
323 
324     /*
325      * TSXデータ走査第一パス(プロパティ取得)
326      */
327     for (var line = 0; line < SrcData.length; line++) {
328         /*
329          * 本体情報の確認
330          * レイヤカウント・各レイヤの継続時間カウント
331          * タイムシートの長さは最長のレイヤを使用
332          * シートの継続はサポート=(直後のレイヤと連結)
333          * 空白行はすべてフレームカウント
334          * 開始行および読み込み停止行の直後の行のみ情報行として使用
335          *    第二フィールドを
336          * 継続時間に加算されないデータ    /^[\/eE].*$/
337          */
338         if (SrcData[line].match(/^[\/eE].*$/)) {
339             if (LayerCount != SrcData.layerCount) {
340                 SrcData.layers[LayerCount].duration = LayerDuration;
341                 LayerDuration = 0;
342                 LayerCount++;
343             }
344             /*
345              * 記述終了・継続時間加算リセット・レイヤ加算
346              */
347         } else {
348             if (LayerCount == SrcData.layerCount) {
349                 SrcData.layerCount++;
350                 SrcData.layers[LayerCount] = {};
351                 SrcData.layers[LayerCount].blmtd = "file";
352                 SrcData.layers[LayerCount].blpos = "first";
353                 SrcData.layers[LayerCount].lot = "=AUTO=";
354             }
355             LayerDuration++;
356             if (SrcData.time < LayerDuration) {
357                 SrcData.time = LayerDuration;
358             }
359         }
360         //有効な動画番号データ(単独)	/^[1-9][0-9]*\s?.*$/
361         //有効な動画番号データ(繰返)	/^[+rR]?\[?[1-9][\,]\]?
362     }
363     /**
364      * プロパティの設定
365      * @type {string}
366      */
367     SrcData.mapfile = "(no file)";
368     SrcData.title = myTitle;
369     SrcData.subtitle = mySubTitle;
370     SrcData.opus = myOpus;
371     SrcData.scene = "";
372     SrcData.cut = "";
373     SrcData.create_user = myName;
374     SrcData.update_user = myName;
375     SrcData.create_time = "";
376     SrcData.update_time = "";
377 
378     SrcData.memo = "converted from TSX data";
379 
380     SrcData.framerate = "";
381 
382     SrcData.trin = [0, "trin"];
383     SrcData.trout = [0, "trout"];
384 
385     /*
386      * 第一パス終了・読み取った情報でXPSオブジェクトを初期化
387      */
388     SrcData.duration = SrcData.time;//TSXはトランジションを扱わない
389     /*
390      * 仮オブジェクトにマッピングしてストリームで返す
391      */
392     var myXps = new Xps(SrcData.layerCount, SrcData.time);
393 
394 
395     /*
396      * 第一パスで読み取ったプロパティをXPSに転記
397      */
398     myXps.mapfile = SrcData.mapfile;
399     myXps.title = SrcData.title;
400     myXps.subtitle = SrcData.subtitle;
401     myXps.opus = SrcData.opus;
402     myXps.scene = SrcData.scene;
403     myXps.cut = SrcData.cut;
404     myXps.create_user = SrcData.create_user;
405     myXps.update_user = SrcData.update_user;
406     myXps.create_time = SrcData.create_time;
407     myXps.update_time = SrcData.update_time;
408 
409 //    myXps.memo = SrcData.memo;
410     myXps.xpsTracks.noteText = SrcData.memo;
411 
412     myXps.framerate = SrcData.framerate;
413 
414     myXps.trin = SrcData.trin;
415     myXps.trout = SrcData.trout;
416 
417     /*
418      * 読み取りデータを調べて得たキーメソッドとブランク位置を転記
419      */
420     for (var lyr = 0; lyr < SrcData.layers.length; lyr++) {
421         var trackID=lyr+1;
422         myXps.xpsTracks[trackID].blmtd = SrcData.layers[lyr].blmtd;
423         myXps.xpsTracks[trackID].blpos = SrcData.layers[lyr].blpos;
424         myXps.xpsTracks[trackID].lot = SrcData.layers[lyr].lot;
425     }
426 
427     /*
428      * TSXデータ走査第二パス(タイムライン取得)
429      */
430 
431     /*
432      * カウンタ初期化
433      */
434     SrcData.time = 0;//初期化
435     var LayerTime = 0;
436     SrcData.layerCount = 0;//初期化
437     LayerCount = 0;
438     var RepeatBuf = [];
439     var repIdx = 0;
440     /*
441      * 本体データ読み取り
442      */
443     for (var line = 0; line < SrcData.length; line++) {
444 
445         if (SrcData[line].match(/^[\/eE].*$/)) {
446             if (LayerCount != SrcData.layerCount) {
447                 LayerTime = 0;
448                 LayerCount++;
449             }
450             /*
451              * 記述終了・継続時間加算リセット・レイヤ加算
452              */
453         } else {
454             if (LayerCount == SrcData.layerCount) {
455                 if (RepeatBuf.length) {
456                     RepeatBuf.length = 0;
457                     repIdx = 0;
458                 }
459                 SrcData.layerCount++;
460             }
461             body_data = SrcData[line].replace(/^([^\#]*)(\-\-|\#).*$/, "$1");
462             if (body_data.match(/^[1-9][0-9]*$/)) {
463                 if (RepeatBuf.length) {
464                     RepeatBuf.length = 0;
465                     repIdx = 0;
466                 }
467                 myXps.xpsTrcks[LayerCount + 1][LayerTime] = body_data;
468             } else {
469                 if (body_data == "") {
470                     if (RepeatBuf.length) {
471                         myXps.xpsTrcks[LayerCount + 1][LayerTime] = RepeatBuf[repIdx % RepeatBuf.length];
472                         repIdx++;
473                     } else {
474                         myXps.xpsTrcks[LayerCount + 1][LayerTime] = body_data;
475                     }
476                 } else {
477                     RepeatBuf = TSX_expdList(body_data);
478                     repIdx = 0;
479                     myXps.xpsTrcks[LayerCount + 1][LayerTime] = RepeatBuf[repIdx];
480                     repIdx++;
481                 }
482 
483             }
484             LayerTime++;
485             if (SrcData.time < LayerDuration) {
486                 SrcData.time = LayerDuration;
487             }
488         }
489 
490     }
491     return myXps.toString();
492 }
