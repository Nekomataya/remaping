1 ﻿/**
  2  * @overview object nas.otome.systemWatcher(myCommand)
  3  * 
  4  * systemWatcherオブジェクトはシステムのクリップボードと特定のフォルダを監視するシステムを提供する。
  5  * 設定されたインターバルで指定フォルダおよびクリップボードの監視を行うことができる。
  6  * 外部プログラムをエージェントとしてAE（乙女）に変更を通知することも可能。
  7  * これらをトリガーにしてコマンドキューに登録したアクションを実行することができる
  8  * 
  9  * アクションにはFunction またはFileオブジェクトを指定する。
 10  * アクションはキューに複数登録しておくことが可能
 11  * 
 12  * 引数:myComamnd にはコマンド文字列としてステータスを表す文字列または設定するインターバルを与える
 13  * [整数]	監視インターバルをミリ秒で指定。デフォルトは　60000 (1分間隔)
 14  * start	監視を開始する。初期化手順が終了するとこのステータスは自動消滅する。
 15  * stop	ステータスフラグを停止にする。AEは呼び出されてもアクションを行なわない。このフラグを認識したエージェントは動作を停止することが望まれる（　停止しなくても良い　）AEの動作はsleepとほぼ同じ
 16  * sleep	ステータスフラグを休止にする。監視継続中だがAEは呼び出されてもアクションを行なわない。エージェントはAEを呼び出さないことが望まれる。外部から呼び出しを行う前にステータスを確認して必要なら変更をトライすること。トライした変更は保証されない。
 17  * wakeup	ステータスフラグを活動中にする。エージェントはターゲットを監視してクリップボードやファイルが更新されたらAEを呼び出す。AEは呼び出しを受けるとキューのコマンドを逐次実行する。
 18  * 
 19  * 監視プロセス実行中は、ステータス遷移コマンドは、以下のように働く
 20  * start	監視プロセスを初期化する。初期化終了後は自身のステータスを動作設定にしたがってsleep/wakeupのいずれかに変更する。
 21  * stop	監視プロセスを終了ステータスへ遷移する。監視インターバルは停止する。外部からの呼び出しは無視する。startupコマンドのみがこの状態を稼動状態へ遷移させる
 22  * sleep	wakeup状態の監視プロセスがこのコマンドを受け取ると、sleep状態に遷移する。インターバル監視は停止。外部からの呼び出しはwakup/stop以外のコマンドを受け取らない
 23  * wakeup	sleep状態の監視プロセスはこのコマンドを受け取ると、wakeup状態に遷移する。監視インターバルは設定された間隔でクリップボードとフォルダの監視を行う。外部呼出しはすべてのコマンドを受信する。
 24  * 
 25  * 
 26  * 監視エージェントの起動は別プロセスにゆだねる。ステータスと一元管理をしない
 27  * 手作業でエージェントを起動しても良い。
 28  * 
 29  * エージェント監視フォルダにあるコマンドフォルダを参照されたし。このフォルダの名前をトリガーにしてエージェントは動作をコントロールすることを期待されている。
 30  * 指定できるコマンド（フォルダ名）は以下の三つのうちいずれか。
 31  * 上から順に優先でそのフォルダが存在すればＡＥ側のステータスがその状態である
 32  * stop.cmd 	エージェント終了
 33  * sleep.cmd	通知停止
 34  * wakeup.cmd	通知開始
 35  * 
 36  * エージェントは自己の希望する動作状態をotomeに通知する。　この通知を受けたAEは、可能な限りステータスを変更してコマンドフォルダを名称変更する。
 37  * フォルダの名前を外部から手作業で変更することでエージェントの動作をコントロールすることも可能だが、エージェントはAEに対して変更リクエストを出すことを望まれる。
 38  * エージェントはAEに問い合わせを行なって現在の監視フォルダのロケーションを得ることが出来るようにする（固定コーディングしない）
 39  *

216 /**
217  * nas.otome.sysyemWatcher.commandQueue
218  * 
219  * コマンドキュー
220  * 監視オブジェクトのコマンドコレクション
221  * コマンドコレクションに登録するのはコマンドオブジェクト
222  * コマンドの比較メソッドが必要かも。重複したコマンドを削除する必要がありそう　何しろ同じ操作をみなおこないがち。
223  * ボタン二回クリックとか
224  * 基本的には、システムとしては完全に重複したものだけを削除する機能を提供する。
225  * それ以外はユーザ判断で処理
226  * 比較メソッドはコマンドオブジェクトのメンバ関数
227  *
228  * @param myTarget
229  * @param watchFile
230  * @param isEnabled
231  * @param myFunc
232  */
233 nas.watchCommand = function (myTarget, watchFile, isEnabled, myFunc) {
234     if (!myTarget) {
235         myTarget = "none"
236     }
237     //"folder""clipboard""both"or"none"　任意の文字列でも可
238     if (!watchFile) {
239         watchFile = false
240     }
241     //FilePath(url) or File ?コマンド実行時のスキップ用
242     if (!isEnabled) {
243         isEnabled = false
244     }
245     //実行保留フラグ
246     if (!myFunc) {
247         myFunc = function () {
248             alert(this.parent.watchFile)
249         }
250     }
251     //コマンド
252     this.watchFile = watchFile;
253     this.watchTarget = myTarget;
254     this.enabled = isEnabled;
255     this.targetCommand = myFunc;
256     this.isSame = function (myCommand) {
257         //コマンド同士の比較　完全一致の場合のみtrue enabledは一時変数の性格が強いので比較対象外に
258         if (this.watchFile.fsName != myCommand.watchFile.fsName) {
259             return false
260         }
261         if (this.watchTarget != myCommand.watchTarget) {
262             return false
263         }
264         if (this.targetCommand.toSource() != myCommand.targetCommand.toSource()) {
265             return false
266         }
267         return true;
268     }
269 };

281     if (!myTarget) {
282         myTarget = "none"
283     }
284     //"folder""clipboard""both"
285     if (!watchFile) {

289     if (!isEnabled) {

291     }
292     //待機(実行保留)フラグ(bool) true / false　フラグが立っている間はキューされても当該コマンドを実行しない　削除は削除メソッドを使用のこと
293     if (!myFunc) {

295     }

300         if (this.commandQueue[ix].isSame(newCommand)) {
301             return false
302         }
303     }
304     this.commandQueue.push(newCommand);
305     return this.commandQueue.length - 1;

402 /**
403  * @param myTarget
404  */
405 nas.otome.systemWatcher.notice = function (myTarget) {
406 
407 //notice 受信タイミングでフォルダのステータスを更新
408     this.checkStatus();
409     //ターゲット指定フォルダがあるかどうか確認する。
410     var folderContent = this.watchFolder.getFiles("*.notice");
411     var myTargetOrder = false;
412     if (folderContent.length) {
413         myTargetOrder = folderContent[0].name.replace(/\.notice$/, "");
414         for (var idf = 0; idf < folderContent.length; idf++) {
415             folderContent[idf].remove()
416         }
417     }
418     if (this.status.match(/stop|sleep/i)) {
419         return
420     }
421     //notice を受信してもstatusがstop sleepならば無条件で処理中止
422     if (!myTarget) {
423         if (myTargetOrder) {
424             myTarget = myTargetOrder
425         } else {
426             myTarget = ""
427         }
428     }
429     //ターゲットオーダーファイルがあればターゲット未指定のターゲットを置き換え
430     nas.otome.writeConsole(myTarget + " notice!");
431     
432 //コマンドキューを逐次処理。相関はなし
433 //呼び出された
434     for (var xidx = 0; xidx < this.commandQueue.length; xidx++) {
435         if (!this.commandQueue[xidx].enabled) {
436             continue
437         }
438         /**
439          * 有効で無い場合は動作スキップ
440          * @type {boolean}
441          */
442         var exF = false;
443         /**
444          * 監視ターゲットを比較　一致または"all"
445          */
446         if ((myTarget == this.commandQueue[xidx].watchTarget) || (this.commandQueue[xidx].watchTarget == "all")) {
447             exF = true
448         }
449         /**
450          * 監視対象ファイルが登録されていてファイルが存在しない場合は動作をスキップ
451          */
452         if (this.commandQueue[xidx].watchFile) {
453 
454             if (this.commandQueue[xidx].watchFile instanceof File) {
455                 var tempFile = this.commandQueue[xidx].watchFile;
456             } else {
457                 var tempFile = new File(this.commandQueue[xidx].watchFile);
458             }
459             if (!tempFile.exists) {
460                 exF = false;
461             }
462         }
463         if (exF) {
464             this.commandQueue[xidx].targetCommand()
465         }
466         //コマンド実行
467     }
468 };
