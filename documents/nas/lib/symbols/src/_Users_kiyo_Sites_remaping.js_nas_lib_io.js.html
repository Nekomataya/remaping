 21 /**
 22  * りまぴん専用AEキー生成関数
 23  * 与える引数は、単一レイヤから導いた1次元配列
 24  * インデックス0から連続してすべてのフレームに値があること。
 25  * プロパティ .label にラベル名を持っても良い。なければダミーを生成する。
 26  *
 27  * 将来、データツリー構造が拡張された場合、機能開始時点でツリーの仮構築必須
 28  * 現在は、決め打ち
 29  * 内部処理に必要な環境を作成
 30  *
 31  * @param layer_id
 32  * @returns {*}
 33  * @private
 34  */
 35 function mkAEKey_(layer_id) {
 36     var layerDataArray = this[layer_id + 1];
 37     layerDataArray.label = this.layers[layer_id].name;
 38 
 39     var blank_method = this.layers[layer_id].blmtd;
 40     var blank_pos = this.layers[layer_id].blpos;
 41     var key_method = xUI.keyMethod;
 42     var key_max_lot = (isNaN(this.layers[layer_id].lot)) ?
 43         0 : this.layers[layer_id].lot;
 44 
 45     /**
 46      * ブランク処理フラグ
 47      * @type {boolean}
 48      */
 49     var bflag = (blank_pos) ? false : true;
 50     var AE_version = xUI.aeVersion;
 51     var compFramerate = this.framerate;
 52     var footageFramerate = xUI.fpsF;
 53     if (isNaN(footageFramerate)) {
 54         footageFramerate = compFramerate
 55     }
 56     var sizeX = this.layers[layer_id].sizeX;
 57     var sizeY = this.layers[layer_id].sizeY;
 58     var aspect = this.layers[layer_id].aspect;
 59     //alert("カラセル方式は :"+blank_method+"\n フーテージのフレームレートは :"+footageFramerate);
 60 
 61     /**
 62      * レイヤロット変数の初期化
 63      * @type {number}
 64      */
 65     var layer_max_lot = 0;
 66 
 67 //前処理 シート配列からキー変換前にフルフレーム有効データの配列を作る
 68 
 69     /**
 70      * 全フレーム分のバッファ配列を作る
 71      * @type {Array}
 72      */
 73     var bufDataArray = new Array(layerDataArray.length);
 74     //キースタック配列を宣言
 75 
 76     /**
 77      * キースタックは可変長
 78      * @type {Array}
 79      */
 80     var keyStackArray = [];
 81     keyStackArray["remap"] = [];
 82     keyStackArray["blank"] = [];
 83     //ふたつ リマップキー/ブランクキー 用
 84 
 85     /**
 86      * 第一フレーム評価・エントリが無効な場合空フレームを設定
 87      * @type {string}
 88      */
 89     bufDataArray[0] = (dataCheck(layerDataArray[0], layerDataArray.label, bflag)) ?
 90         dataCheck(layerDataArray[0], layerDataArray.label, bflag) : "blank";
 91 
 92     /**
 93      * 2～ラストフレームループ
 94      */
 95     for (f = 1; f < layerDataArray.length; f++) {
 96         //有効データを判定して無効データエントリを直前のコピーで埋める
 97         bufDataArray[f] = (dataCheck(layerDataArray[f], layerDataArray.label, bflag)) ?
 98             dataCheck(layerDataArray[f], layerDataArray.label, bflag) : bufDataArray[f - 1];
 99 
100         if (bufDataArray[f] != "blank") {
101             layer_max_lot = (layer_max_lot > bufDataArray[f]) ?
102                 layer_max_lot : bufDataArray[f];
103         }
104     }
105     max_lot = (layer_max_lot > key_max_lot) ?
106         layer_max_lot : key_max_lot;
107 
108     /**
109      * あらかじめ与えられた最大ロット変数と有効データ中の最大の値を比較して
110      * 大きいほうをとる
111      * ここで、layer_max_lot が 0 であった場合変換すべきデータが無いので処理中断
112      */
113     if (layer_max_lot == 0) {
114         return "変換すべきデータがありません。\n処理を中断します。";
115     }
116 
117 //前処理第二 (配列には、キーを作成するフレームを積む)
118 
119     keyStackArray["remap"].push(0);
120     keyStackArray["blank"].push(0);//最初のフレームには無条件でキーを作成
121 
122     /**
123      * 有効データで埋まった配列を再評価(2～ラスト)
124      */
125     for (f = 1; f < bufDataArray.length; f++) {
126 //キーオプションにしたがって以下の評価でキー配列にスタック(フレームのみ)
127         switch (key_method) {
128             case    "opt"    :	//	最適化キー(変化点の前後にキー)
129                 //	○前データと同じで、かつ後ろのデータと
130                 //	同一のエントリをスキップ
131                 if (bufDataArray[f] != bufDataArray[f - 1] || bufDataArray[f] != bufDataArray[f + 1]) {
132                     keyStackArray["remap"].push(f)
133                 }
134                 break;
135             case    "min"    :	//	最少キー(変化点の前後にキー)
136                 //	○前データと同じエントリをスキップ
137                 if (bufDataArray[f] != bufDataArray[f - 1]) {
138                     keyStackArray["remap"].push(f)
139                 }
140                 break;
141             case    "max"    :	//	全フレームキー(スキップ無し)
142             default:
143                 keyStackArray["remap"].push(f);
144         }
145         /**
146          * ブランクメソッドにしたがってブランクキーをスタック(フレームのみ)
147          * @type {string}
148          */
149         var prevalue = (bufDataArray[f - 1] == "blank") ? "blank" : "cell";
150         var currentvalue = (bufDataArray[f] == "blank") ? "blank" : "cell";
151         var postvalue = (bufDataArray[f + 1] == "blank") ? "blank" : "cell";
152         switch (key_method) {
153             case    "opt"    :	//	最適化キー(変化点の前後にキー)
154                 if (currentvalue != prevalue || currentvalue != postvalue) {
155                     keyStackArray["blank"].push(f)
156                 }
157                 break;
158             case    "min"    :	//	最少キー(変化点の前後にキー)
159                 if (currentvalue != prevalue) {
160                     keyStackArray["blank"].push(f)
161                 }
162                 break;
163             case    "max"    :	//	全フレームキー(スキップ無し)
164             default:
165                 keyStackArray["blank"].push(f);
166         }
167     }
168 
169     /**
170      * キー文字列を作成
171      * blankoffsetは、カラセル挿入によるタイミングの遷移量・冒頭挿入以外は基本的に0
172      */
173     switch (blank_pos) {
174         case    "first"    :
175             var blankoffset = 1;
176             break;
177         case    "end"    :
178             var blankoffset = 0;
179             break;
180         case    "none"    :
181             var blankoffset = 0;
182             break;
183         default    :
184             var blankoffset = 0;
185     }
186     var footage_frame_duration = (1 / footageFramerate);
187     /**
188      * リマップキーを作成
189      * @type {string}
190      */
191     var remapBody =
192         'Time Remap\n\tFrame\tseconds\t\n';
193     for (n = 0; n < keyStackArray["remap"].length; n++) {
194         if (bufDataArray[keyStackArray["remap"][n]] == "blank") {
195             var seedValue = (blank_pos == "first") ? 1 : max_lot + 1;
196         } else {
197             var seedValue = bufDataArray[keyStackArray["remap"][n]] * 1 + blankoffset;
198         }
199         remapBody += "\t";
200         remapBody += keyStackArray["remap"][n].toString(10);
201         remapBody += "\t";
202         if (blank_method == "expression2" &&
203             bufDataArray[keyStackArray["remap"][n]] == "blank") {
204             remapBody += 999999;//エクスプレッション2のカラ
205         } else {
206             remapBody += (seedValue - 0.5) * footage_frame_duration;//通常処理
207         }
208         remapBody += "\t\n";
209     }
210 
211     /**
212      * エクスプレッション型
213      * @type {string}
214      */
215     var expBody =
216         'スライダ\tスライダ制御\tEffect\ Parameter\ #1\t\n\tFrame\t\t\n';
217 
218     for (n = 0; n < keyStackArray["remap"].length; n++) {
219         expBody += "\t";
220         expBody += keyStackArray["remap"][n].toString(10);
221         expBody += "\t";
222         if (bufDataArray[keyStackArray["remap"][n]] == "blank") {
223             expBody += ("0");
224 
225         } else {
226             expBody += (bufDataArray[keyStackArray["remap"][n]]);
227         }
228         expBody += "\t\n";
229     }
230 
231     /**
232      * ブランクキーを作成
233      * エクスプレッション型/ブランクセル無し の場合は不要
234      */
235     switch (blank_method) {
236         case "opacity":
237             /**
238              * 不透明度
239              * @type {string}
240              */
241             var blankBody =
242                 'Opacity\n\tFrame\tpercent\t\n';
243             var blank_ = '0';
244             var cell_ = '100';
245             break;
246         case "wipe":
247             /**
248              * ワイプ
249              * @type {string}
250              */
251             var blankBody =
252                 '変換終了\tリニアワイプ\tEffect\ Parameter\ \#1\t\n\tFrame\tパーセント\t\n';
253             var blank_ = '100';
254             var cell_ = '0';
255             break;
256     }
257 
258     for (n = 0; n < keyStackArray["blank"].length; n++) {
259 
260         blankBody += "\t";
261         blankBody += keyStackArray["blank"][n].toString(10);
262         blankBody += "\t";
263         if (bufDataArray[keyStackArray["blank"][n]] == "blank") {
264             blankBody += blank_;
265         } else {
266             blankBody += cell_;
267         }
268 
269         blankBody += "\t\n";
270     }
271 
272     /**
273      * 出力
274      * @type {string}
275      */
276     var Result = 'Adobe After Effects ';
277     Result += AE_version;
278     Result += ' Keyframe Data\n';
279 //Result += ;
280     Result += '\n\tUnits Per Second\t';
281     Result += compFramerate.toString();
282     Result += '\n\tSource Width\t';
283     Result += sizeX.toString();
284     Result += '\n\tSource Height\t';
285     Result += sizeY.toString();
286     Result += '\n\tSource Pixel Aspect Ratio\t1';
287     Result += '\n\tComp Pixel Aspect Ratio\t1';
288     Result += '\n';
289     if (blank_method != "expression1") {
290 
291         if (blank_method == "opacity") {
292             /**
293              * ブランク
294              * @type {string}
295              */
296             Result += '\n';
297             Result += blankBody;
298         }
299         /**
300          * リマップ
301          * @type {string}
302          */
303         Result += '\n';
304         Result += remapBody;
305         if (blank_method == "wipe") {
306             /**
307              * ブランク
308              * @type {string}
309              */
310             Result += '\n';
311             Result += blankBody;
312         }
313 
314     } else {
315         /**
316          * エクスプレッション1
317          * @type {string}
318          */
319         Result += '\n';
320         Result += expBody;
321     }
322 
323     Result += '\n\n';
324     Result += 'End of Keyframe Data';
325 
326     return Result;
327 }

330 /**
331  * リスト展開プロシージャ
332  * @type {boolean}
333  */
334 var expd_repFlag = false;
335 /**
336  * グローバルで宣言
337  * @type {number}
338  */
339 var expd_skipValue = 0;
340 /**
341  * リスト展開はxUIのメソッドか?
342  * @param ListStr
343  * @param rcl
344  * @returns {*}
345  */
346 function nas_expdList(ListStr, rcl) {
347     if (!rcl) {
348         rcl = false
349     } else {
350         rcl = true
351     }
352     var leastCount = (xUI.Selection[1]) ? xUI.Selection[1] : XPS.duration() - xUI.Select[1];
353     if (!rcl) {
354         expd_repFlag = false;
355         expd_skipValue = xUI.spinValue - 1;
356         //再帰呼び出し以外はスピン値で初期化
357     }
358     /**
359      * (スキップ量はスピン-１)この値はグローバルの値を参照
360      * @type {string}
361      */
362     var SepChar = "\.";
363 //	var SepChars=["\.","\ ","\,"];
364 
365 
366     /**
367      *  台詞レイヤの場合のみ、カギ括弧の中をすべてセパレートする
368      */
369     if (xUI.Select[0] == 0) {
370         if (ListStr.match(/「([^「]*)」?/)) ;
371         if (ListStr.match(/「(.+)」?/)) {
372             //alert("Hit ："+ListStr.match(/^(.*「)([^」]*)(」?$)/));
373             ListStr = d_break(ListStr.match(/^(.*「)([^」]*)(」?$)/));
374             ListStr = ListStr.replace(/「/g, SepChar + "「" + SepChar);//開き括弧はセパレーション
375         }
376         ListStr = ListStr.replace(/\」/g, "---");//閉じ括弧は横棒
377         ListStr = ListStr.replace(/\、/g, "・");//読点中黒
378         ListStr = ListStr.replace(/\。/g, "");//句点空白(null)
379         ListStr = ListStr.replace(/\ー/g, "｜");//音引き縦棒
380     }
381 
382     /**
383      * 導入リピートならば専用展開プロシージャへ渡してしまう
384      */
385     if (ListStr.match(/^([\+rR])(.*)$/)) {
386         var expdList = TSX_expdList(ListStr);
387         expd_repFlag = true;
388     } else {
389 
390         /**
391          * リスト文字列を走査してセパレータを置換
392          * @type {XML|string|void}
393          */
394         ListStr = ListStr.replace(/[\,\x20]/g, SepChar);
395         /**
396          * スラッシュを一組で括弧と置換(代用括弧)
397          * @type {XML|string}
398          */
399         ListStr = ListStr.replace(/\/(.*)(\/)/g, "\($1\)");//コメント引数注意
400 //		var PreX="/\(\.([1-9])/g";//括弧の前にセパレータを補う
401         ListStr = ListStr.replace(/\(([^\.])/g, "\(\.$1");
402 //		var PostX="/[0-9](\)[1-9])/";//括弧の後にセパレータを補う
403         ListStr = ListStr.replace(/([^\.])(\)[1-9]?)/g, "$1\.$2");
404 
405         /**
406          * 前処理終わり
407          */
408 
409         /**
410          * リストをセパレータで分割して配列に
411          * @type {Array}
412          */
413         var srcList = [];
414         srcList = ListStr.toString().split(SepChar);
415 
416 //	if (xUI.Select[0]==0){
417 //	} else {}
418 
419         /**
420          * 生成データ配列を作成
421          * @type {Array}
422          */
423         var expdList = [];
424 
425         var sDepth = 0;//括弧展開深度/初期値0
426         var StartCt = 0;
427         var EndCt = 0;
428 //rT=0;
429         /**
430          * 元配列を走査
431          */
432 
433         /**
434          * ローカルスコープにするために宣言する
435          * @type {number}
436          */
437         var ct = 0;
438         for (ct = 0; ct < srcList.length; ct++) {
439             tcn = srcList[ct];
440 
441             /**
442              * トークンが開きカギ括弧の場合リザルトに積まないで
443              * リザルトのおしまいの要素を横棒にする。
444              */
445 
446             //	if (tcn.match(/^[「」]$/)) {y_bar();continue;}
447             if (tcn == "「") {
448                 y_bar();
449                 continue;
450             }
451 
452             /**
453              * トークンがコントロールワードならば値はリザルトに積まない
454              * 変数に展開してループ再開
455              */
456             if (tcn.match(/^s([1-9][0-9]*)$/)) {
457                 // if(RegExp.$1*1>0) {xUI.spin(RegExp.$1*1)}else{xUI.spin(1)};
458                 expd_skipValue = (RegExp.$1 * 1 > 0) ? (RegExp.$1 * 1 - 1) : 0;
459                 continue;
460             }
461 
462             /**
463              * グローバル
464              */
465 
466             /**
467              * トークンが開き括弧ならばデプスを加算して保留
468              */
469             if (tcn.match(/^(\(|\/)$/)) {
470                 sDepth = 1;
471                 StartCt = ct;
472 
473                 /**
474                  * トークンを積まないで閉じ括弧を走査
475                  */
476 
477                 /**
478                  * ローカルスコープにするために宣言する
479                  * @type {number}
480                  */
481                 var ct2 = 0;
482                 for (ct2 = ct + 1; ct2 < srcList.length; ct2++) {
483                     if (srcList[ct2].match(/^\($/)) {
484                         sDepth++
485                     }
486                     if (srcList[ct2].match(/^(\)|\/)[\*x]?([0-9]*)$/)) {
487                         sDepth--
488                     }
489                     if (sDepth == 0) {
490                         EndCt = ct2;
491                         /**
492                          * 最初の括弧が閉じたので括弧の繰り返し分を取得/ループ
493                          * @type {number}
494                          */
495                         var rT = RegExp.$2 * 1;
496                         if (rT < 1) {
497                             rT = 1;
498                         }
499 
500                         if (RegExp.$2 == "") {
501                             expd_repFlag = true;
502                         }
503 
504                         /**
505                          * ローカルスコープにするために宣言する
506                          * @type {number}
507                          */
508                         var ct3 = 0;
509                         for (ct3 = 1; ct3 <= rT; ct3++) {
510                             if ((StartCt + 1) != EndCt) {
511                                 // alert("DPS= "+sDepth+" :start= "+StartCt+"  ;end= "+EndCt +"\n"+ srcList.slice(StartCt+1,EndCt).join(SepChar)+"\n\n-- "+rT);
512                                 expdList = expdList.concat(nas_expdList(srcList.slice(StartCt + 1, EndCt).join(SepChar), "Rcall"));
513 
514                                 /**
515                                  * 括弧の中身を自分自身に渡して展開させる
516                                  * 展開配列が規定処理範囲を超過していたら処理終了
517                                  */
518                                 if (expdList.length >= leastCount) {
519                                     return expdList
520                                 }
521                             }
522                         }
523                         ct = EndCt;
524                         break;
525                     }//if block end
526                 }//ct2 loop end
527 
528                 if (rT == 0) {
529                     expdList.push(srcList[ct]);
530                     s_kip();//ct++;
531                 }
532             } else {
533                 // トークンが展開可能なら展開して生成データに積む
534                 if (tcn.match(/^([1-9]{1}[0-9]*)\-([1-9]{1}[0-9]*)$/)) {
535                     var stV = Math.round(RegExp.$1 * 1);
536                     var edV = Math.round(RegExp.$2 * 1);
537                     if (stV <= edV) {
538                         for (tcv = stV; tcv <= edV; tcv++) {
539                             expdList.push(tcv);
540                             s_kip();
541                         }
542                     } else {
543                         for (tcv = stV; tcv >= edV; tcv--) {
544                             expdList.push(tcv);
545                             s_kip();
546                         }
547                     }
548                 } else {
549                     expdList.push(tcn);
550                     s_kip();
551                 }
552 
553             }
554         }
555     }
556     /**
557      * 生成配列にスキップを挿入
558      */
559     function s_kip() {
560         // for (x=0;x<(xUI.spinValue-1);x++){expdList.push('');}
561         for (x = 0; x < expd_skipValue; x++) {
562             expdList.push('');
563         }
564     }
565 
566     /**
567      * 配列の末尾を横棒に
568      */
569     function y_bar() {
570         expdList.pop();
571         expdList.push('---');
572     }
573 
574     /**
575      * かぎ括弧の中身をセパレーション
576      * @param dList
577      * @returns {*}
578      */
579     function d_break(dList) {
580         wLists = dList.toString().split(",");
581         return wLists[1] + wLists[2].replace(/(.)/g, "$1\.") + wLists[3];
582     }
583 
584 
585     /**
586      * カエス
587      */
588     if (expdList.length < leastCount && expd_repFlag) {
589         blockCount = expdList.length;
590         // alert(blockCount + " / " +leastCount);
591         for (resultCt = 0; resultCt <= (leastCount - blockCount); resultCt++) {
592             expdList.push(expdList[resultCt % blockCount]);
593         }
594     }
595     return expdList;
596 
597 }
