 61 /**
 62  * @desc ftgFolderを初期化する
 63  * 順次以前のコードを修正すること
 64  */
 65 nas.ftgFolders.init = function () {
 66     this.length = -1;//ftgBaseを除外するために-1から開始
 67     this.cpRx = [];//比較配列を作る
 68     this.names = [];//比較現状のフォルダ名配列
 69     for (var prp in this) {
 70         if (!( prp.match(/(init|length|cpRx|names)/) )) {
 71             //各プロパティの正規表現を先に組んでおく
 72             var myRegStr = this[prp].join("|");
 73             this.cpRx.push(new RegExp("\(" + myRegStr + "\)", "i"));//正規表現を要素にする
 74             this.names.push(this[prp][0]);//アイテム名をリストしておく
 75             this.length++;
 76         }
 77     }
 78 };

227 nas.inputMedias = new nTable();
228 nas.inputMedias.onChange = function () {
229     /**
230      * メディアがセレクトされたらシステムの解像度とフレームレートを変更する
231      */
232     if (nas.LENGTH != this.selectedRecord[1]) {
233         nas.LENGTH = this.selectedRecord[1]
234     }
235     if (nas.ASPECT != this.selectedRecord[2]) {
236         nas.ASPECT = this.selectedRecord[2]
237     }
238     var myDPC = this.selectedRecord[3] / 2.540;//解像度をDPCに変換
239     if (nas.RESOLUTION != myDPC) {
240         nas.RESOLUTION = myDPC
241     }
242     if (nas.FRATE.rate != this.selectedRecord[4]) {
243         nas.FRATE = nas.newFramerate("",this.selectedRecord[4]);
244     }
245 
246     nas.registerMarks.select(this.selectedRecord[5]);
247 };
