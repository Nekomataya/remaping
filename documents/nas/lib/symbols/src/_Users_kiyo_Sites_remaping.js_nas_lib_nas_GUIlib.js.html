266 nas.GUI.Grid = function (col, line, width, height, parent) {

269     top = (line * this.lineUnit) + this.topMargin * M + this.topPadding;
270     right = left * 1 + width * this.colUnit - this.rightPadding - this.leftPadding;
271     bottom = (height <= this.lineUnit) ?

273

275     return [left, top, right, bottom];
